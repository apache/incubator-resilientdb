55std::unique_ptr<std::string> KVServerExecutor::ExecuteData(
56 const std::string& request) {
57 KVRequest kv_request;
58 KVResponse kv_response;
59
60 if (!kv_request.ParseFromString(request)) {
61 LOG(ERROR) << "parse data fail";
62 return nullptr;
63 }
64
65 if (kv_request.cmd() == KVRequest::SET) {
66 Set(kv_request.key(), kv_request.value());
67 } else if (kv_request.cmd() == KVRequest::GET) {
68 kv_response.set_value(Get(kv_request.key()));
69 } else if (kv_request.cmd() == KVRequest::GETVALUES) {
70 kv_response.set_value(GetValues());
71 } else if (kv_request.cmd() == KVRequest::GETRANGE) {
72 kv_response.set_value(GetRange(kv_request.key(), kv_request.value()));
73 }
74
75 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
76 if (!kv_response.SerializeToString(resp_str.get())) {
77 return nullptr;
78 }
79
80 return resp_str;
81}

83// Validate transactions committed by the Python SDK
84bool KVServerExecutor::Validate(const std::string& transaction) {
85 auto locals = py::dict("transaction"_a = transaction);
86
87 py::exec(R"(
88 from sdk_validator.validator import is_valid_tx
89
90 try:
91 txn_dict = json.loads(transaction)
92 ret = is_valid_tx(txn_dict)
93 is_valid = ret[0] == 0
94 except (KeyError, AttributeError, ValueError):
95 is_valid = False
96 )",
97 py::globals(), locals);
98
99 bool is_valid = locals["is_valid"].cast<bool>();
100
101 return is_valid;
102}

104void KVServerExecutor::Set(const std::string& key, const std::string& value) {
105 if (require_txn_validation_) {
106 bool is_valid = Validate(value);
107 if (!is_valid) {
108 LOG(ERROR) << "Invalid transaction for " << key;
109 return;
110 }
111 }
112
113 if (equip_rocksdb_) {
114 r_storage_layer_.SetValue(key, value);
115 } else if (equip_leveldb_) {
116 l_storage_layer_.SetValue(key, value);
117 } else {
118 kv_map_[key] = value;
119 }
120}

122std::string KVServerExecutor::Get(const std::string& key) {
123 if (equip_rocksdb_) {
124 return r_storage_layer_.GetValue(key);
125 } else if (equip_leveldb_) {
126 return l_storage_layer_.GetValue(key);
127 } else {
128 auto search = kv_map_.find(key);
129 if (search != kv_map_.end())
130 return search->second;
131 else
132 return "";
133 }
134}

136std::string KVServerExecutor::GetValues() {
137 if (equip_rocksdb_) {
138 return r_storage_layer_.GetAllValues();
139 } else if (equip_leveldb_) {
140 return l_storage_layer_.GetAllValues();
141 } else {
142 std::string values = "[";
143 bool first_iteration = true;
144 for (auto kv : kv_map_) {
145 if (!first_iteration) values.append(",");
146 first_iteration = false;
147 values.append(kv.second);
148 }
149 values.append("]");
150 return values;
151 }
152}

154// Get values on a range of keys
155std::string KVServerExecutor::GetRange(const std::string& min_key,
156 const std::string& max_key) {
157 if (equip_rocksdb_) {
158 return r_storage_layer_.GetRange(min_key, max_key);
159 } else if (equip_leveldb_) {
160 return l_storage_layer_.GetRange(min_key, max_key);
161 } else {
162 std::string values = "[";
163 bool first_iteration = true;
164 for (auto kv : kv_map_) {
165 if (kv.first >= min_key && kv.first <= max_key) {
166 if (!first_iteration) values.append(",");
167 first_iteration = false;
168 values.append(kv.second);
169 }
170 }
171 values.append("]");
172 return values;
173 }
