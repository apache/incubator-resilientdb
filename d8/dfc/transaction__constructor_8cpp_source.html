34absl::StatusOr<std::string> TransactionConstructor::GetResponseData(
35 const Response& response) {
36 std::string hash_;
37 std::set<int64_t> hash_counter;
38 std::string resp_str;
39 for (const auto& each_resp : response.resp()) {
40 // Check signature
41 std::string hash = SignatureVerifier::CalculateHash(each_resp.data());
42
43 if (!hash_.empty() && hash != hash_) {
44 LOG(ERROR) << "hash not the same";
45 return absl::InvalidArgumentError("hash not match");
46 }
47 if (hash_.empty()) {
48 hash_ = hash;
49 resp_str = each_resp.data();
50 }
51 hash_counter.insert(each_resp.signature().node_id());
52 }
53 // LOG(INFO) << "recv hash:" << hash_counter.size()
54 // << " need:" << config_.GetMinClientReceiveNum()
55 // << " data len:" << resp_str.size();
56 if (hash_counter.size() >=
57 static_cast<size_t>(config_.GetMinClientReceiveNum())) {
58 return resp_str;
59 }
60 return absl::InvalidArgumentError("data not enough");
61}

70int TransactionConstructor::SendRequest(
71 const google::protobuf::Message& message,
72 google::protobuf::Message* response, Request::Type type) {
73 NetChannel::SetDestReplicaInfo(config_.GetReplicaInfos()[0]);
74 int ret = NetChannel::SendRequest(message, type, true);
75 if (ret == 0) {
76 std::string resp_str;
77 int ret = NetChannel::RecvRawMessageData(&resp_str);
78 if (ret >= 0) {
79 if (!response->ParseFromString(resp_str)) {
80 LOG(ERROR) << "parse response fail:" << resp_str.size();
81 return -2;
82 }
83 return 0;
84 }
85 }
86 return -1;
87}
