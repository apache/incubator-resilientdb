104// handle the response message. If receive f+1 commit messages, send back to the
105// client.
106int PerformanceManager::ProcessResponseMsg(std::unique_ptr<Context> context,
107 std::unique_ptr<Request> request) {
108 std::unique_ptr<Request> response;
109 // Add the response message, and use the call back to collect the received
110 // messages.
111 // The callback will be triggered if it received f+1 messages.
112 if (request->ret() == -2) {
113 // LOG(INFO) << "get response fail:" << request->ret();
114 send_num_--;
115 return 0;
116 }
117 CollectorResultCode ret =
118 AddResponseMsg(context->signature, std::move(request),
119 [&](const Request& request,
120 const TransactionCollector::CollectorDataType*) {
121 response = std::make_unique<Request>(request);
122 return;
123 });
124
125 if (ret == CollectorResultCode::STATE_CHANGED) {
126 BatchClientResponse batch_response;
127 if (batch_response.ParseFromString(response->data())) {
128 SendResponseToClient(batch_response);
129 } else {
130 LOG(ERROR) << "parse response fail:";
131 }
132 }
133 return ret == CollectorResultCode::INVALID ? -2 : 0;
134}

136bool PerformanceManager::MayConsensusChangeStatus(
137 int type, int received_count, std::atomic<TransactionStatue>* status) {
138 switch (type) {
139 case Request::TYPE_RESPONSE:
140 // if receive f+1 response results, ack to the client.
141 if (*status == TransactionStatue::None &&
142 config_.GetMinClientReceiveNum() <= received_count) {
143 TransactionStatue old_status = TransactionStatue::None;
144 return status->compare_exchange_strong(
145 old_status, TransactionStatue::EXECUTED, std::memory_order_acq_rel,
146 std::memory_order_acq_rel);
147 }
148 break;
149 }
150 return false;
151}

153CollectorResultCode PerformanceManager::AddResponseMsg(
154 const SignatureInfo& signature, std::unique_ptr<Request> request,
155 std::function<void(const Request&,
156 const TransactionCollector::CollectorDataType*)>
157 response_call_back) {
158 if (request == nullptr) {
159 return CollectorResultCode::INVALID;
160 }
161
162 int type = request->type();
163 uint64_t seq = request->seq();
164 int resp_received_count = 0;
165 int ret = collector_pool_->GetCollector(seq)->AddRequest(
166 std::move(request), signature, false,
167 [&](const Request& request, int received_count,
168 TransactionCollector::CollectorDataType* data,
169 std::atomic<TransactionStatue>* status) {
170 if (MayConsensusChangeStatus(type, received_count, status)) {
171 resp_received_count = 1;
172 response_call_back(request, data);
173 }
174 });
175 if (ret != 0) {
176 return CollectorResultCode::INVALID;
177 }
178 if (resp_received_count > 0) {
179 collector_pool_->Update(seq);
180 return CollectorResultCode::STATE_CHANGED;
181 }
182 return CollectorResultCode::OK;
183}

185void PerformanceManager::SendResponseToClient(
186 const BatchClientResponse& batch_response) {
187 uint64_t create_time = batch_response.createtime();
188 uint64_t local_id = batch_response.local_id();
189 if (create_time > 0) {
190 uint64_t run_time = get_sys_clock() - create_time;
191 global_stats_->AddLatency(run_time);
192 } else {
193 LOG(ERROR) << "seq:" << local_id << " no resp";
194 }
195 send_num_--;
196
197 if (config_.IsPerformanceRunning()) {
198 return;
199 }
200}

203int PerformanceManager::BatchProposeMsg() {
204 LOG(WARNING) << "batch wait time:" << config_.ClientBatchWaitTimeMS()
205 << " batch num:" << config_.ClientBatchNum()
206 << " max txn:" << config_.GetMaxProcessTxn();
207 std::vector<std::unique_ptr<QueueItem>> batch_req;
208 eval_ready_future_.get();
209 while (!stop_) {
210 if (send_num_ > config_.GetMaxProcessTxn()) {
211 usleep(100000);
212 continue;
213 }
214 if (batch_req.size() < config_.ClientBatchNum()) {
215 std::unique_ptr<QueueItem> item =
216 batch_queue_.Pop(config_.ClientBatchWaitTimeMS());
217 if (item == nullptr) {
218 continue;
219 }
220 batch_req.push_back(std::move(item));
221 if (batch_req.size() < config_.ClientBatchNum()) {
222 continue;
223 }
224 }
225 int ret = DoBatch(batch_req);
226 batch_req.clear();
227 if (ret != 0) {
228 Response response;
229 response.set_result(Response::ERROR);
230 for (size_t i = 0; i < batch_req.size(); ++i) {
231 if (batch_req[i]->context && batch_req[i]->context->client) {
232 int ret = batch_req[i]->context->client->SendRawMessage(response);
233 if (ret) {
234 LOG(ERROR) << "send resp" << response.DebugString()
235 << " fail ret:" << ret;
236 }
237 }
238 }
239 }
240 }
241 return 0;
242}

244int PerformanceManager::DoBatch(
245 const std::vector<std::unique_ptr<QueueItem>>& batch_req) {
246 auto new_request =
247 NewRequest(Request::TYPE_NEW_TXNS, Request(), config_.GetSelfInfo().id());
248 if (new_request == nullptr) {
249 return -2;
250 }
251 std::vector<std::unique_ptr<Context>> context_list;
252
253 BatchClientRequest batch_request;
254 for (size_t i = 0; i < batch_req.size(); ++i) {
255 BatchClientRequest::ClientRequest* req =
256 batch_request.add_client_requests();
257 *req->mutable_request() = *batch_req[i]->client_request.get();
258 if (batch_req[i]->context) {
259 *req->mutable_signature() = batch_req[i]->context->signature;
260 }
261 req->set_id(i);
262 }
263
264 batch_request.set_createtime(get_sys_clock());
265 batch_request.SerializeToString(new_request->mutable_data());
266 if (verifier_) {
267 auto signature_or = verifier_->SignMessage(new_request->data());
268 if (!signature_or.ok()) {
269 LOG(ERROR) << "Sign message fail";
270 return -2;
271 }
272 *new_request->mutable_data_signature() = *signature_or;
273 }
274
275 new_request->set_hash(SignatureVerifier::CalculateHash(new_request->data()));
276 new_request->set_proxy_id(config_.GetSelfInfo().id());
277
278 replica_client_->SendMessage(*new_request, GetPrimary());
279 global_stats_->BroadCastMsg();
280 send_num_++;
281 if (total_num_++ == 1000000) {
282 stop_ = true;
283 LOG(WARNING) << "total num is done:" << total_num_;
284 }
285 if (total_num_ % 10000 == 0) {
286 LOG(WARNING) << "total num is :" << total_num_;
287 }
288 global_stats_->IncClientCall();
289 return 0;
290}
