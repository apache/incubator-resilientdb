30 * TransactionManager is an interface to manager the transactions that have been
31 * committed by the replicas. ExecuteData will be called inside ExecuteBatch to
32 * execute each transaction. One of the interface ExecuteBatch or ExecuteData
33 * should be implemented, but not both.
34 */
35class TransactionManager {
36 public:
37 TransactionManager(bool is_out_of_order = false, bool need_response = true);
38 virtual ~TransactionManager() = default;

40 virtual std::unique_ptr<BatchUserResponse> ExecuteBatch(
41 const BatchUserRequest& request);
42
43 virtual std::unique_ptr<std::string> ExecuteData(const std::string& request);
44
45 bool IsOutOfOrder();
46
47 bool NeedResponse();
48
49 virtual Storage* GetStorage() { return nullptr; };
50
51 private:

54};
