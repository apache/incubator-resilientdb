35int RecvInternal(int fd, void* buf, size_t len) {
36 size_t pos = 0;
37 while (len > 0) {
38 int ret = recv(fd, (char*)buf + pos, len, 0);
39 if (ret <= 0) {
40 // LOG(ERROR) << "recv data fail, fd =" << fd << " error " <<
41 // strerror(errno)
42 // << " ret:" << ret;
43 return ret;
44 }
45 pos += ret;
46 len -= ret;
47 }
48 return pos;
49}

51int SendInternal(int fd, const void* buf, size_t len) {
52 if (fd < 0) {
53 return -2;
54 }
55 size_t pos = 0;
56 while (pos < len) {
57 int ret = send(fd, (char*)buf + pos, len - pos, 0);
58 if (ret < 0) {
59 LOG(ERROR) << "send data fail, fd =" << fd << " error "
60 << strerror(errno);
61 return -1;
62 }
63 pos += ret;
64 }
65 return pos;
66}

97int TcpSocket::Listen(const std::string& ip, int port) {
98 int on = 1;
99 if (setsockopt(socket_fd_, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) != 0) {
100 LOG(ERROR) << "set TcpSocket opt fail, error" << strerror(errno);
101 return -1;
102 }
103
104 struct sockaddr_in servaddr;
105 memset(&servaddr, 0, sizeof(servaddr));
106 servaddr.sin_family = AF_INET;
107 servaddr.sin_addr.s_addr = inet_addr(ip.data());
108 servaddr.sin_port = htons(port);
109
110 if (bind(socket_fd_, (struct sockaddr*)&servaddr, sizeof(servaddr)) == -1) {
111 LOG(ERROR) << "bind TcpSocket error: " << strerror(errno)
112 << "(errno: " << errno << ")";
113 return -1;
114 }
115
116 if (listen(socket_fd_, 10) == -1) {
117 LOG(ERROR) << "listen TcpSocket error: " << strerror(errno)
118 << "(errno: " << errno << ")";
119 return -1;
120 }
121 if (port == 0) {
122 struct sockaddr_in localaddr;
123 socklen_t len = sizeof(localaddr);
124 int ret = getsockname(socket_fd_, (struct sockaddr*)&localaddr, &len);
125 if (ret != 0) {
126 LOG(ERROR) << "get binding port fail:" << strerror(errno);
127 } else {
128 binding_port_ = ntohs(localaddr.sin_port);
129 }
130 } else {
131 binding_port_ = port;
132 }
133 return 0;
134}

177int TcpSocket::Send(const std::string& data) {
178 size_t data_size = data.size();
179 size_t send_ret = SendInternal(socket_fd_, &data_size, sizeof(data_size));
180 if (send_ret != sizeof(data_size)) {
181 if (static_cast<int>(send_ret) == -1) {
182 LOG(ERROR) << "send data error: " << strerror(errno)
183 << "(errno: " << errno << ")";
184 return -1;
185 }
186 return -2;
187 }
188
189 send_ret = SendInternal(socket_fd_, data.c_str(), data_size);
190 if (send_ret != data_size) {
191 if (static_cast<int>(send_ret) == -1) {
192 LOG(ERROR) << "send data not done: " << strerror(errno)
193 << "(errno: " << errno << ")";
194 return -1;
195 } else {
196 return -2;
197 }
198 }
199 return 0;
200}

202int TcpSocket::Recv(void** buf, size_t* len) {
203 int ret = RecvInternal(socket_fd_, len, sizeof(size_t));
204 if (ret <= 0) {
205 return ret;
206 }
207 *buf = malloc(*len);
208 ret = RecvInternal(socket_fd_, *buf, *len);
209 if (ret <= 0) {
210 return ret;
211 }
212 return ret;
213}
