32int UTXOClient::Transfer(const UTXO& utxo) {
33 UTXORequest request;
34 UTXOResponse response;
35 *request.mutable_utxo() = utxo;
36
37 int ret = SendRequest(request, &response);
38 if (ret != 0 || response.ret() < 0) {
39 return -1;
40 }
41 return response.ret();
42}

44std::vector<UTXO> UTXOClient::GetList(int64_t end_id, int num) {
45 UTXOQuery query;
46 query.set_query_transaction(true);
47 query.set_end_id(end_id);
48 query.set_num(num);
49
50 CustomQueryResponse response;
51
52 int ret = SendRequest(query, Request::TYPE_CUSTOM_QUERY);
53 if (ret) {
54 LOG(ERROR) << "send request fail";
55 return std::vector<UTXO>();
56 }
57
58 ret = RecvRawMessage(&response);
59 if (ret) {
60 LOG(ERROR) << "recv response fail";
61 return std::vector<UTXO>();
62 }
63
64 UTXOQueryResponse utxo_response;
65 utxo_response.ParseFromString(response.resp_str());
66 std::vector<UTXO> utxo_list;
67 for (const auto& utxo : utxo_response.utxos()) {
68 utxo_list.push_back(utxo);
69 }
70 return utxo_list;
71}
