20def read_tps(file):
21 tps = []
22 lat = []
23 with open(file) as f:
24 for l in f.readlines():
25 s = l.split()
26 for r in s:
27 if(r.split(':')[0] == 'txn'):
28 tps.append(int(r.split(':')[1]))
29 if l.find("client latency") > 0:
30 lat.append(float(s[-1].split(':')[-1]))
31 return tps, lat
