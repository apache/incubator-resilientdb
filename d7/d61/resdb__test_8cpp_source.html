75class ResDBTest : public Test {

79 void StartOneServer(int server_id) {
80 ResDBConfig config = GetConfig(server_id);
81 config.SetTestMode(true);
82 auto executor = std::make_unique<ResDBTestManager>(config);
83 executors_.push_back(executor.get());
84 server_.push_back(std::make_unique<ServiceNetwork>(
85 config,
86 std::make_unique<ConsensusManagerPBFT>(config, std::move(executor))));
87 server_thread_.push_back(std::thread(
88 [&](ServiceNetwork* server) { server->Run(); }, server_.back().get()));
89 }
90
91 void WaitAllServerStarted() {
92 for (auto& server : server_) {
93 while (true) {
94 std::unique_lock<std::mutex> lk(mutex_);
95 cv_.wait_for(lk, std::chrono::microseconds(1000),
96 [&] { return server->ServiceIsReady(); });
97 if (server->ServiceIsReady()) {
98 break;
99 }
100 }
101 }
102 }
103
104 void WaitExecutorDone(int received_num) {
105 for (auto executor : executors_) {
106 while (static_cast<int>(executor->GetSeqs().size()) < received_num) {
107 usleep(10000);
108 }
109 }
110 }
111
112 void Stop() {
113 for (auto& server : server_) {
114 server->Stop();
115 }
116 for (auto& th : server_thread_) {
117 th.join();
118 }
119 server_.clear();
120 }
