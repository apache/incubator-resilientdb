160def _fulfillment_to_details(fulfillment):
161 """Encode a fulfillment as a details dictionary
162
163 Args:
164 fulfillment: Crypto-conditions Fulfillment object
165 """
166
167 if fulfillment.type_name == 'ed25519-sha-256':
168 return {
169 'type': 'ed25519-sha-256',
170 'public_key': base58.b58encode(fulfillment.public_key).decode(),
171 }
172
173 if fulfillment.type_name == 'threshold-sha-256':
174 subconditions = [
175 _fulfillment_to_details(cond['body'])
176 for cond in fulfillment.subconditions
177 ]
178 return {
179 'type': 'threshold-sha-256',
180 'threshold': fulfillment.threshold,
181 'subconditions': subconditions,
182 }
183
184 raise UnsupportedTypeError(fulfillment.type_name)

187def _fulfillment_from_details(data, _depth=0):
188 """Load a fulfillment for a signing spec dictionary
189
190 Args:
191 data: tx.output[].condition.details dictionary
192 """
193 if _depth == 100:
194 raise ThresholdTooDeep()
195
196 if data['type'] == 'ed25519-sha-256':
197 public_key = base58.b58decode(data['public_key'])
198 return Ed25519Sha256(public_key=public_key)
199
200 if data['type'] == 'threshold-sha-256':
201 threshold = ThresholdSha256(data['threshold'])
202 for cond in data['subconditions']:
203 cond = _fulfillment_from_details(cond, _depth+1)
204 threshold.add_subfulfillment(cond)
205 return threshold
206
207 raise UnsupportedTypeError(data.get('type'))

359 def generate(cls, public_keys, amount):
360 """Generates a Output from a specifically formed tuple or list.
361
362 Note:
363 If a ThresholdCondition has to be generated where the threshold
364 is always the number of subconditions it is split between, a
365 list of the following structure is sufficient:
366
367 [(address|condition)*, [(address|condition)*, ...], ...]
368
369 Args:
370 public_keys (:obj:`list` of :obj:`str`): The public key of
371 the users that should be able to fulfill the Condition
372 that is being created.
373 amount (:obj:`int`): The amount locked by the Output.
374
375 Returns:
376 An Output that can be used in a Transaction.
377
378 Raises:
379 TypeError: If `public_keys` is not an instance of `list`.
380 ValueError: If `public_keys` is an empty list.
381 """
382 threshold = len(public_keys)
383 if not isinstance(amount, int):
384 raise TypeError('`amount` must be a int')
385 if amount < 1:
386 raise AmountError('`amount` needs to be greater than zero')
387 if not isinstance(public_keys, list):
388 raise TypeError('`public_keys` must be an instance of list')
389 if len(public_keys) == 0:
390 raise ValueError('`public_keys` needs to contain at least one'
391 'owner')
392 elif len(public_keys) == 1 and not isinstance(public_keys[0], list):
393 if isinstance(public_keys[0], Fulfillment):
394 ffill = public_keys[0]
395 else:
396 ffill = Ed25519Sha256(
397 public_key=base58.b58decode(public_keys[0]))
398 return cls(ffill, public_keys, amount=amount)
399 else:
400 initial_cond = ThresholdSha256(threshold=threshold)
401 threshold_cond = reduce(cls._gen_condition, public_keys,
402 initial_cond)
403 return cls(threshold_cond, public_keys, amount=amount)

405 @classmethod
406 def _gen_condition(cls, initial, new_public_keys):
407 """Generates ThresholdSha256 conditions from a list of new owners.
408
409 Note:
410 This method is intended only to be used with a reduce function.
411 For a description on how to use this method, see
412 :meth:`~.Output.generate`.
413
414 Args:
415 initial (:class:`cryptoconditions.ThresholdSha256`):
416 A Condition representing the overall root.
417 new_public_keys (:obj:`list` of :obj:`str`|str): A list of new
418 owners or a single new owner.
419
420 Returns:
421 :class:`cryptoconditions.ThresholdSha256`:
422 """
423 try:
424 threshold = len(new_public_keys)
425 except TypeError:
426 threshold = None
427
428 if isinstance(new_public_keys, list) and len(new_public_keys) > 1:
429 ffill = ThresholdSha256(threshold=threshold)
430 reduce(cls._gen_condition, new_public_keys, ffill)
431 elif isinstance(new_public_keys, list) and len(new_public_keys) <= 1:
432 raise ValueError('Sublist cannot contain single owner')
433 else:
434 try:
435 new_public_keys = new_public_keys.pop()
436 except AttributeError:
437 pass
438 # NOTE: Instead of submitting base58 encoded addresses, a user
439 # of this class can also submit fully instantiated
440 # Cryptoconditions. In the case of casting
441 # `new_public_keys` to a Ed25519Fulfillment with the
442 # result of a `TypeError`, we're assuming that
443 # `new_public_keys` is a Cryptocondition then.
444 if isinstance(new_public_keys, Fulfillment):
445 ffill = new_public_keys
446 else:
447 ffill = Ed25519Sha256(
448 public_key=base58.b58decode(new_public_keys))
449 initial.add_subfulfillment(ffill)
450 return initial

689 @classmethod
690 def transfer(cls, inputs, recipients, asset_id, metadata=None):
691 """A simple way to generate a `TRANSFER` transaction.
692
693 Note:
694 Different cases for threshold conditions:
695
696 Combining multiple `inputs` with an arbitrary number of
697 `recipients` can yield interesting cases for the creation of
698 threshold conditions we'd like to support. The following
699 notation is proposed:
700
701 1. The index of a `recipient` corresponds to the index of
702 an input:
703 e.g. `transfer([input1], [a])`, means `input1` would now be
704 owned by user `a`.
705
706 2. `recipients` can (almost) get arbitrary deeply nested,
707 creating various complex threshold conditions:
708 e.g. `transfer([inp1, inp2], [[a, [b, c]], d])`, means
709 `a`'s signature would have a 50% weight on `inp1`

711 weight respectively. `inp2` is owned completely by `d`.
712
713 Args:

715 Input`): Converted `Output`s, intended to

717 recipients (:obj:`list` of :obj:`tuple`): A list of
718 ([keys],amount) that represent the recipients of this
719 Transaction.

721 this Transaction.

723 Transaction.
724
725 Returns:
726 :class:`~resdb_validator.transaction.Transaction`
727 """
