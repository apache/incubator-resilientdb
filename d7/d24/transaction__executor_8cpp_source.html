87int TransactionExecutor::Commit(std::unique_ptr<Request> message) {
88 global_stats_->IncPendingExecute();
89 if (transaction_manager_ && transaction_manager_->IsOutOfOrder()) {
90 // LOG(ERROR)<<"add out of order exe:"<<message->seq()<<" from

92 std::unique_ptr<Request> msg = std::make_unique<Request>(*message);
93 execute_OOO_queue_.Push(std::move(message));
94 commit_queue_.Push(std::move(msg));

105std::unique_ptr<Request> TransactionExecutor::GetNextData() {
106 if (candidates_.empty() || candidates_.begin()->first != next_execute_seq_) {
107 return nullptr;
108 }
109 auto res = std::move(candidates_.begin()->second);
110 if (pre_exec_func_) {
111 pre_exec_func_(res.get());
112 }
113 candidates_.erase(candidates_.begin());
114 return res;
115}

117void TransactionExecutor::OrderMessage() {
118 while (!IsStop()) {
119 auto message = commit_queue_.Pop();
120 if (message != nullptr) {
121 global_stats_->IncExecute();
122 uint64_t seq = message->seq();
123 if (next_execute_seq_ > seq) {
124 // LOG(INFO) << "request seq:" << seq << " has been executed"
125 // << " next seq:" << next_execute_seq_;
126 continue;
127 }
128
129 AddNewData(std::move(message));
130 }
131
132 while (!IsStop()) {
133 std::unique_ptr<Request> message = GetNextData();
134 if (message == nullptr) {
135 break;
136 }
137 execute_queue_.Push(std::move(message));
138 next_execute_seq_++;
139 if (seq_update_notify_func_) {
140 seq_update_notify_func_(next_execute_seq_);
141 }
142 }
143 }
144 return;
145}

147void TransactionExecutor::ExecuteMessage() {
148 while (!IsStop()) {
149 auto message = execute_queue_.Pop();
150 if (message == nullptr) {
151 continue;
152 }
153 bool need_execute = true;
154 if (transaction_manager_ && transaction_manager_->IsOutOfOrder()) {
155 need_execute = false;
156 }
157 Execute(std::move(message), need_execute);
158 }
159}

199void TransactionExecutor::Execute(std::unique_ptr<Request> request,
200 bool need_execute) {
201 // Execute the request, then send the response back to the user.
202 BatchUserRequest batch_request;
203 if (!batch_request.ParseFromString(request->data())) {
204 LOG(ERROR) << "parse data fail";
205 }
206 batch_request.set_seq(request->seq());
207 batch_request.set_hash(request->hash());
208 batch_request.set_proxy_id(request->proxy_id());
209 if (request->has_committed_certs()) {
210 *batch_request.mutable_committed_certs() = request->committed_certs();
211 }
212
213 // LOG(INFO) << " get request batch size:"
214 // << batch_request.user_requests_size()<<" proxy
215 // id:"<<request->proxy_id()<<" need execute:"<<need_execute;
216 // std::unique_ptr<BatchUserResponse> batch_response =
217 // std::make_unique<BatchUserResponse>();
218
219 std::unique_ptr<BatchUserResponse> response;
220 if (transaction_manager_ && need_execute) {
221 response = transaction_manager_->ExecuteBatch(batch_request);
222 }
223
224 if (duplicate_manager_) {
225 duplicate_manager_->AddExecuted(batch_request.hash(), batch_request.seq());
226 }
227
228 global_stats_->IncTotalRequest(batch_request.user_requests_size());
229 if (response == nullptr) {
230 response = std::make_unique<BatchUserResponse>();
231 }
232
233 response->set_createtime(batch_request.createtime());
234 response->set_local_id(batch_request.local_id());
235 response->set_hash(batch_request.hash());
236
237 post_exec_func_(std::move(request), std::move(response));
238
239 global_stats_->IncExecuteDone();
240}
