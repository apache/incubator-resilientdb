48bool ED25519verifyString(const std::string& message,
49 const std::string& public_key,
50 const std::string& signature) {
51 CryptoPP::byte byteKey[CryptoPP::ed25519PrivateKey::PUBLIC_KEYLENGTH];
52 if (public_key.size() != CryptoPP::ed25519PrivateKey::PUBLIC_KEYLENGTH) {
53 LOG(ERROR) << "public key len invalid:" << public_key.size();
54 return false;
55 }
56 memcpy(byteKey, public_key.c_str(), public_key.size());
57 auto verifier = CryptoPP::ed25519::Verifier(byteKey);
58
59 bool valid = true;
60 CryptoPP::StringSource(
61 signature + message, true,
62 new CryptoPP::SignatureVerificationFilter(
63 verifier,
64 new CryptoPP::ArraySink((CryptoPP::byte*)&valid, sizeof(valid))));
65 if (!valid) {
66 LOG(ERROR) << "signature invalid. signature len:" << signature.size()
67 << " message len:" << message.size();
68 }
69 return valid;
70}

72bool CmacVerifyString(const std::string& message, const std::string& public_key,
73 const std::string& signature) {
74 bool res = false;
75 // KEY TRANSFORMATION
76 // https://stackoverflow.com/questions/26145776/string-to-secbyteblock-conversion
77 CryptoPP::SecByteBlock privKey((const unsigned char*)(public_key.data()),
78 public_key.size());
79 CryptoPP::CMAC<CryptoPP::AES> cmac(privKey.data(), privKey.size());
80 const int flags = CryptoPP::HashVerificationFilter::HASH_AT_END |
81 CryptoPP::HashVerificationFilter::PUT_RESULT;
82
83 // MESSAGE VERIFICATION
84 CryptoPP::StringSource ss3(
85 message + signature, true,
86 new CryptoPP::HashVerificationFilter(
87 cmac, new CryptoPP::ArraySink((CryptoPP::byte*)&res, sizeof(res)),
88 flags)); // StringSource
89 return res;
90}

105// CMAC-AES Signature generator
106// Return a token, mac, to verify the a message.
107std::string CmacSignString(const std::string& private_key,
108 const std::string& message) {
109 std::string mac = "";
110
111 // KEY TRANSFORMATION.
112 // https://stackoverflow.com/questions/26145776/string-to-secbyteblock-conversion
113
114 CryptoPP::SecByteBlock mac_private_key(
115 (const unsigned char*)(private_key.data()), private_key.size());
116
117 CryptoPP::CMAC<CryptoPP::AES> cmac(mac_private_key.data(),
118 mac_private_key.size());
119
120 CryptoPP::StringSource ss1(
121 message, true,
122 new CryptoPP::HashFilter(cmac, new CryptoPP::StringSink(mac)));
123 return mac;
124}

128SignatureVerifier::SignatureVerifier(const KeyInfo& private_key,
129 const CertificateInfo& certificate_info) {
130 private_key_ = private_key;
131 if (private_key_.key().empty()) {
132 LOG(ERROR) << "private invalid";
133 }
134
135 node_id_ = certificate_info.node_id();
136 admin_public_key_ = certificate_info.admin_public_key();
137 AddPublicKey(certificate_info.public_key(), false);
138 if (private_key_.hash_type() == SignatureInfo::ED25519) {
139 CryptoPP::byte byteKey[CryptoPP::ed25519PrivateKey::SECRET_KEYLENGTH];
140 if (private_key_.key().size() !=
141 CryptoPP::ed25519PrivateKey::SECRET_KEYLENGTH) {
142 return;
143 }
144 memcpy(byteKey, private_key_.key().c_str(), private_key_.key().size());
145 signer_ = std::make_unique<CryptoPP::ed25519::Signer>(byteKey);
146 }
147}

149bool SignatureVerifier::AddPublicKey(const CertificateKey& public_key,

151 std::unique_lock<std::shared_mutex> lk(mutex_);
152 if (need_verify &&

185// Funtion to calculate hash of a string.
186std::string SignatureVerifier::CalculateHash(const std::string& str) {
187 CryptoPP::byte const* pData = (CryptoPP::byte*)str.data();
188 unsigned int nDataLen = str.size();
189 CryptoPP::byte aDigest[CryptoPP::SHA256::DIGESTSIZE];
190
191 CryptoPP::SHA256().CalculateDigest(aDigest, pData, nDataLen);
192 return std::string((char*)aDigest, CryptoPP::SHA256::DIGESTSIZE);
193}

195absl::StatusOr<SignatureInfo> SignatureVerifier::SignMessage(
196 const std::string& message) {
197 SignatureInfo info;
198 info.set_node_id(node_id_);
199 switch (private_key_.hash_type()) {
200 case SignatureInfo::RSA:
201 info.set_signature(utils::RsaSignString(private_key_.key(), message));
202 break;
203 case SignatureInfo::ED25519:
204 info.set_signature(ED25519signString(message, signer_.get()));
205 break;
206 case SignatureInfo::CMAC_AES:
207 info.set_signature(CmacSignString(private_key_.key(), message));
208 break;
209 case SignatureInfo::ECDSA:
210 info.set_signature(utils::ECDSASignString(private_key_.key(), message));
211 break;
212 default:
213 break;
214 }
215 return info;
216}

259bool SignatureVerifier::VerifyMessage(const std::string& message,
260 const KeyInfo& public_key,
261 const std::string& signature) {
262 if (public_key.key().empty() || signature.empty()) {
263 LOG(ERROR) << "public is empty, size(" << public_key.key().size()
264 << ") or signature is empty, size(" << signature.size() << ")";
265 return false;
266 }
267 // LOG(ERROR) << "public key hash type:" << public_key.hash_type()
268 // << " key size:" << public_key.key().size()
269 // << " sig size:" << signature.size()
270 // << " msg size:" << message.size();
271 switch (public_key.hash_type()) {
272 case SignatureInfo::RSA:
273 return utils::RsaVerifyString(message, public_key.key(), signature);
274 case SignatureInfo::ED25519:
275 return ED25519verifyString(message, public_key.key(), signature);
276 case SignatureInfo::CMAC_AES:
277 return CmacVerifyString(message, public_key.key(), signature);
278 case SignatureInfo::ECDSA:
279 return utils::ECDSAVerifyString(message, public_key.key(), signature);
280 default:
281 return true;
282 }
283}
