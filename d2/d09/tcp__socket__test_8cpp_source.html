30TEST(TcpSocket, SendAndRecv) {
31 std::promise<bool> svr_done, cli_done;
32 std::future<bool> svr_done_future = svr_done.get_future(),
33 cli_done_future = cli_done.get_future();
34
35 std::thread svr_thead = std::thread([&]() {
36 TcpSocket svr_socket;
37 svr_socket.Listen("127.0.0.1", 1234);
38 svr_done.set_value(true);
39 auto client_socket = svr_socket.Accept();
40 char *buf = nullptr;
41 size_t len = 0;
42 int ret = client_socket->Recv((void **)&buf, &len);
43 EXPECT_EQ(ret, 4);
44 EXPECT_EQ(std::string(buf, len), "test");
45 free(buf);
46
47 ret = client_socket->Send("recv_suc");
48 EXPECT_EQ(ret, 0);
49
50 cli_done_future.get();
51 });
52
53 svr_done_future.get();
54 // Sever is ready, start the client.
55
56 TcpSocket client_socket;
57 int ret = client_socket.Connect("127.0.0.1", 1234);
58 ASSERT_EQ(ret, 0);
59 ret = client_socket.Send("test");
60 ASSERT_EQ(ret, 0);
61
62 char *buf = nullptr;
63 size_t len = 0;
64 ret = client_socket.Recv((void **)&buf, &len);
65 EXPECT_EQ(ret, 8);
66 EXPECT_EQ(std::string(buf, len), "recv_suc");
67 free(buf);
68
69 cli_done.set_value(true);
70 svr_thead.join();
71}
