76void Stats::MonitorGlobal() {
77 LOG(ERROR) << "monitor:" << name_ << " sleep time:" << monitor_sleep_time_;
78
79 uint64_t seq_fail = 0;
80 uint64_t client_call = 0, socket_recv = 0;
81 uint64_t num_client_req = 0, num_propose = 0, num_prepare = 0, num_commit = 0,
82 pending_execute = 0, execute = 0, execute_done = 0;
83 uint64_t broad_cast_msg = 0, send_broad_cast_msg = 0;
84 uint64_t send_broad_cast_msg_per_rep = 0;
85 uint64_t server_call = 0, server_process = 0;
86 uint64_t seq_gap = 0;
87 uint64_t total_request = 0, total_geo_request = 0, geo_request = 0;
88
89 // ====== for client proxy ======
90 uint64_t run_req_num = 0, run_req_run_time = 0;
91
92 uint64_t last_run_req_num = 0, last_run_req_run_time = 0;
93 // =============================
94
95 uint64_t last_seq_fail = 0;
96 uint64_t last_num_client_req = 0, last_num_propose = 0, last_num_prepare = 0,
97 last_num_commit = 0;
98 uint64_t last_pending_execute = 0, last_execute = 0, last_execute_done = 0;
99 uint64_t last_client_call = 0, last_socket_recv = 0;
100 uint64_t last_broad_cast_msg = 0, last_send_broad_cast_msg = 0;
101 uint64_t last_send_broad_cast_msg_per_rep = 0;
102 uint64_t last_server_call = 0, last_server_process = 0;
103 uint64_t last_total_request = 0, last_total_geo_request = 0,
104 last_geo_request = 0;
105 uint64_t time = 0;
106
107 while (!stop_) {
108 sleep(monitor_sleep_time_);
109 time += monitor_sleep_time_;
110 seq_fail = seq_fail_;
111 socket_recv = socket_recv_;
112 client_call = client_call_;
113 num_client_req = num_client_req_;
114 num_propose = num_propose_;
115 num_prepare = num_prepare_;
116 num_commit = num_commit_;
117 pending_execute = pending_execute_;
118 execute = execute_;
119 execute_done = execute_done_;
120 broad_cast_msg = broad_cast_msg_;
121 send_broad_cast_msg = send_broad_cast_msg_;
122 send_broad_cast_msg_per_rep = send_broad_cast_msg_per_rep_;
123 server_call = server_call_;
124 server_process = server_process_;
125 seq_gap = seq_gap_;
126 total_request = total_request_;
127 total_geo_request = total_geo_request_;
128 geo_request = geo_request_;
129
130 run_req_num = run_req_num_;
131 run_req_run_time = run_req_run_time_;
132
133 LOG(ERROR) << "=========== monitor =========\n"
134 << "server call:" << server_call - last_server_call
135 << " server process:" << server_process - last_server_process
136 << " socket recv:" << socket_recv - last_socket_recv
137 << " "
138 "client call:"
139 << client_call - last_client_call
140 << " "
141 "client req:"
142 << num_client_req - last_num_client_req
143 << " "
144 "broad_cast:"
145 << broad_cast_msg - last_broad_cast_msg
146 << " "
147 "send broad_cast:"
148 << send_broad_cast_msg - last_send_broad_cast_msg
149 << " "
150 "per send broad_cast:"
151 << send_broad_cast_msg_per_rep - last_send_broad_cast_msg_per_rep
152 << " "
153 "propose:"
154 << num_propose - last_num_propose
155 << " "
156 "prepare:"
157 << (num_prepare - last_num_prepare)
158 << " "
159 "commit:"
160 << (num_commit - last_num_commit)
161 << " "
162 "pending execute:"
163 << pending_execute - last_pending_execute
164 << " "
165 "execute:"
166 << execute - last_execute
167 << " "
168 "execute done:"
169 << execute_done - last_execute_done << " seq gap:" << seq_gap
170 << " total request:" << total_request - last_total_request
171 << " txn:" << (total_request - last_total_request) / 5
172 << " total geo request:"
173 << total_geo_request - last_total_geo_request
174 << " total geo request per:"
175 << (total_geo_request - last_total_geo_request) / 5
176 << " geo request:" << (geo_request - last_geo_request)
177 << " "
178 "seq fail:"
179 << seq_fail - last_seq_fail << " time:" << time
180 << " "
181 "\n--------------- monitor ------------";
182 if (run_req_num - last_run_req_num > 0) {
183 LOG(ERROR) << " req client latency:"
184 << static_cast<double>(run_req_run_time -
185 last_run_req_run_time) /
186 (run_req_num - last_run_req_num) / 1000000000.0;
187 }
188
189 last_seq_fail = seq_fail;
190 last_socket_recv = socket_recv;
191 last_client_call = client_call;
192 last_num_client_req = num_client_req;
193 last_num_propose = num_propose;
194 last_num_prepare = num_prepare;
195 last_num_commit = num_commit;
196 last_pending_execute = pending_execute;
197 last_execute = execute;
198 last_execute_done = execute_done;
199
200 last_broad_cast_msg = broad_cast_msg;
201 last_send_broad_cast_msg = send_broad_cast_msg;
202 last_send_broad_cast_msg_per_rep = send_broad_cast_msg_per_rep;
203
204 last_server_call = server_call;
205 last_server_process = server_process;
206
207 last_run_req_num = run_req_num;
208 last_run_req_run_time = run_req_run_time;
209 last_total_request = total_request;
210 last_total_geo_request = total_geo_request;
211 last_geo_request = geo_request;
212 }
213}
