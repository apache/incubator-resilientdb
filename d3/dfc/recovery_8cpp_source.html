105void Recovery::UpdateStableCheckPoint() {
106 if (checkpoint_ == nullptr) {
107 return;
108 }
109 while (!stop_) {
110 int64_t latest_ckpt = checkpoint_->GetStableCheckpoint();
111 LOG(ERROR) << "get stable ckpt:" << latest_ckpt;
112 if (last_ckpt_ == latest_ckpt) {
113 sleep(recovery_ckpt_time_s_);
114 continue;
115 }
116 last_ckpt_ = latest_ckpt;
117 FinishFile(latest_ckpt);
118 }

121void Recovery::GetLastFile() {
122 std::string dir = std::filesystem::path(file_path_).parent_path();
123 last_ckpt_ = -1;
124 int m_time_s = 0;
125 for (const auto& entry : std::filesystem::directory_iterator(dir)) {
126 std::string dir = std::filesystem::path(entry.path()).parent_path();
127 std::string file_name = std::filesystem::path(entry.path()).stem();
128 std::string ext = std::filesystem::path(entry.path()).extension();
129 if (ext != ".log") continue;
130 int pos = file_name.rfind("_");
131 int64_t ckpt = std::stoll(file_name.substr(pos + 1));
132 int max_seq_pos = file_name.rfind("_", pos - 1);
133 int min_seq_pos = file_name.rfind("_", max_seq_pos - 1);
134 int time_pos = file_name.rfind("_", min_seq_pos - 1);
135
136 int64_t min_seq = std::stoll(
137 file_name.substr(min_seq_pos + 1, max_seq_pos - min_seq_pos - 1));
138
139 int64_t time_s =
140 std::stoll(file_name.substr(time_pos + 1, min_seq_pos - time_pos - 1));
141 if (min_seq == -1) {
142 if (last_ckpt_ == -1 || m_time_s < time_s) {
143 file_path_ = entry.path();
144 last_ckpt_ = ckpt;
145 m_time_s = time_s;
146 LOG(ERROR) << "get last path:" << file_name << " min:" << min_seq;
147 }
148 }
149 }
150 if (last_ckpt_ == -1) {
151 last_ckpt_ = 0;
152 file_path_ = GenerateFile(last_ckpt_, -1, -1);
153 }
154}

157 int64_t max_seq) {

159 std::string file_name = std::filesystem::path(base_file_path_).stem();

163 std::to_string(seq);
164 std::string ext = std::filesystem::path(base_file_path_).extension();
165 if (ext == "") ext = "log";
166 return dir + "/" + file_name + "." + ext;

169void Recovery::FinishFile(int64_t seq) {
170 std::unique_lock<std::mutex> lk(mutex_);
171 Flush();
172 if (storage_) {
173 if (!storage_->Flush()) {
174 return;
175 }
176 }
177 std::string new_file_path = GenerateFile(seq, min_seq_, max_seq_);
178 close(fd_);
179
180 min_seq_ = -1;
181 max_seq_ = -1;
182
183 std::rename(file_path_.c_str(), new_file_path.c_str());
184
185 LOG(INFO) << "rename:" << file_path_ << " to:" << new_file_path;
186 std::string next_file_path = GenerateFile(seq, -1, -1);
187 file_path_ = next_file_path;
188
189 OpenFile(file_path_);
190}

192void Recovery::SwitchFile(const std::string& file_path) {
193 std::unique_lock<std::mutex> lk(mutex_);
194
195 min_seq_ = -1;
196 max_seq_ = -1;
197
198 ReadLogsFromFiles(
199 file_path, 0, 0, [&](const SystemInfoData& data) {},
200 [&](std::unique_ptr<Context> context, std::unique_ptr<Request> request) {
201 min_seq_ == -1
202 ? min_seq_ = request->seq()
203 : std::min(min_seq_, static_cast<int64_t>(request->seq()));
204 max_seq_ = std::max(max_seq_, static_cast<int64_t>(request->seq()));
205 });
206
207 OpenFile(file_path);
208 LOG(INFO) << "switch to file:" << file_path << " seq:"
209 << "[" << min_seq_ << "," << max_seq_ << "]";
210}

212void Recovery::OpenFile(const std::string& path) {
213 if (fd_ >= 0) {
214 close(fd_);
215 }
216 fd_ = open(path.c_str(), O_CREAT | O_WRONLY, 0666);
217 if (fd_ < 0) {
218 LOG(ERROR) << "open file fail:" << path << " error:" << strerror(errno);
219 }
220
221 int pos = lseek(fd_, 0, SEEK_END);
222 LOG(INFO) << "file path:" << path << " len:" << pos << " fd:" << fd_;
223
224 if (pos == 0) {
225 WriteSystemInfo();
226 }
227
228 lseek(fd_, 0, SEEK_END);
229 LOG(INFO) << "open file:" << path << " pos:" << lseek(fd_, 0, SEEK_CUR)
230 << " fd:" << fd_;
231 assert(fd_ >= 0);

292std::vector<std::unique_ptr<Recovery::RecoveryData>> Recovery::ParseData(
293 const std::string& data) {
294 std::vector<std::unique_ptr<RecoveryData>> request_list;
295
296 std::vector<std::string> data_list;
297 int pos = 0;
298 while (pos < data.size()) {
299 size_t len;
300 memcpy(&len, data.c_str() + pos, sizeof(len));
301 pos += sizeof(len);
302
303 std::string item = data.substr(pos, len);
304 pos += len;
305 data_list.push_back(item);
306 }
307
308 for (size_t i = 0; i < data_list.size(); i += 2) {
309 std::unique_ptr<RecoveryData> recovery_data =
310 std::make_unique<RecoveryData>();
311 recovery_data->request = std::make_unique<Request>();
312 recovery_data->context = std::make_unique<Context>();
313
314 if (!recovery_data->request->ParseFromString(data_list[i])) {
315 LOG(ERROR) << "Parse from data fail";
316 break;
317 }
318
319 if (!recovery_data->context->signature.ParseFromString(data_list[i + 1])) {
320 LOG(ERROR) << "Parse from data fail";
321 break;
322 }
323
324 request_list.push_back(std::move(recovery_data));
325 }
326 return request_list;
327}

384std::pair<std::vector<std::pair<int64_t, std::string>>, int64_t>
385Recovery::GetRecoveryFiles() {
386 std::string dir = std::filesystem::path(file_path_).parent_path();
387 int64_t last_ckpt = 0;
388 for (const auto& entry : std::filesystem::directory_iterator(dir)) {
389 std::string dir = std::filesystem::path(entry.path()).parent_path();
390 std::string file_name = std::filesystem::path(entry.path()).stem();
391 std::string ext = std::filesystem::path(entry.path()).extension();
392 if (ext != ".log") continue;
393 int pos = file_name.rfind("_");
394 int64_t ckpt = std::stoll(file_name.substr(pos + 1));
395 if (ckpt > last_ckpt) {
396 last_ckpt = ckpt;
397 }
398 }
399 std::vector<std::pair<int64_t, std::string>> list;
400
401 std::vector<std::pair<int64_t, std::string>> e_list;
402
403 for (const auto& entry : std::filesystem::directory_iterator(dir)) {
404 std::string dir = std::filesystem::path(entry.path()).parent_path();
405 std::string file_name = std::filesystem::path(entry.path()).stem();
406 std::string ext = std::filesystem::path(entry.path()).extension();
407 if (ext != ".log") continue;
408 int pos = file_name.rfind("_");
409
410 int max_seq_pos = file_name.rfind("_", pos - 1);
411 int64_t max_seq =
412 std::stoll(file_name.substr(max_seq_pos + 1, pos - max_seq_pos - 1));
413
414 int min_seq_pos = file_name.rfind("_", max_seq_pos - 1);
415 int64_t min_seq = std::stoll(
416 file_name.substr(min_seq_pos + 1, max_seq_pos - min_seq_pos - 1));
417
418 int time_pos = file_name.rfind("_", min_seq_pos - 1);
419 int64_t time =
420 std::stoll(file_name.substr(time_pos + 1, min_seq_pos - time_pos - 1));
421
422 if (min_seq == -1) {
423 e_list.push_back(std::make_pair(time, entry.path()));
424 } else if ((min_seq <= last_ckpt && max_seq >= last_ckpt)) {
425 list.push_back(std::make_pair(time, entry.path()));
426 }
427 }
428
429 sort(e_list.begin(), e_list.end());
430 list.push_back(e_list.back());
431
432 sort(list.begin(), list.end());
433 return std::make_pair(list, last_ckpt);
434}

453void Recovery::ReadLogsFromFiles(
454 const std::string& path, int64_t ckpt, int file_idx,
455 std::function<void(const SystemInfoData& data)> system_callback,
456 std::function<void(std::unique_ptr<Context> context,
457 std::unique_ptr<Request> request)>
458 call_back) {
459 int fd = open(path.c_str(), O_CREAT | O_RDONLY, 0666);
460 if (fd < 0) {
461 LOG(ERROR) << " open file fail:" << path;
462 }
463 LOG(INFO) << "read logs:" << path << " pos:" << lseek(fd, 0, SEEK_CUR);
464 assert(fd >= 0);
465
466 size_t data_len = 0;
467 Read(fd, sizeof(data_len), reinterpret_cast<char*>(&data_len));
468 {
469 std::string data;
470 char* buf = new char[data_len];
471 if (!Read(fd, data_len, buf)) {
472 LOG(ERROR) << "Read system info fail";
473 return;
474 }
475 data = std::string(buf, data_len);
476 delete buf;
477 std::vector<std::string> data_list = ParseRawData(data);
478
479 SystemInfoData info;
480 if (data_list.empty() || !info.ParseFromString(data_list[0])) {
481 LOG(ERROR) << "parse info fail:" << data.size();
482 return;
483 }
484 LOG(INFO) << "read system info:" << info.DebugString();
485 if (file_idx == 0) {
486 system_callback(info);
487 }
488 }
489
490 std::vector<std::unique_ptr<RecoveryData>> request_list;
491
492 while (Read(fd, sizeof(data_len), reinterpret_cast<char*>(&data_len))) {
493 std::string data;
494 char* buf = new char[data_len];
495 if (!Read(fd, data_len, buf)) {
496 LOG(ERROR) << "Read data log fail";
497 break;
498 }
499 data = std::string(buf, data_len);
500 delete buf;
501
502 std::vector<std::unique_ptr<RecoveryData>> list = ParseData(data);
503 if (list.size() == 0) {
504 request_list.clear();
505 break;
506 }
507 for (auto& l : list) {
508 request_list.push_back(std::move(l));
509 }
510 }
511 if (request_list.size() == 0) {
512 ftruncate(fd, 0);
513 }
514 for (std::unique_ptr<RecoveryData>& recovery_data : request_list) {
515 if (ckpt < recovery_data->request->seq()) {
516 recovery_data->request->set_is_recovery(true);
517 call_back(std::move(recovery_data->context),
518 std::move(recovery_data->request));
519 }
520 }
521
522 LOG(INFO) << "read log from files:" << path << " done";
523
524 close(fd);
525}
