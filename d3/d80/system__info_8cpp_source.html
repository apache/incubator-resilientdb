48void SystemInfo::AddReplica(const ReplicaInfo& replica) {
49 if (replica.id() == 0 || replica.ip().empty() || replica.port() == 0) {
50 return;
51 }
52 for (const auto& cur_replica : replicas_) {
53 if (cur_replica.id() == replica.id()) {
54 LOG(ERROR) << " replica exist:" << replica.id();
55 return;
56 }
57 }
58 LOG(ERROR) << "add new replica:" << replica.DebugString();
59 replicas_.push_back(replica);
60}

62void SystemInfo::ProcessRequest(const SystemInfoRequest& request) {
63 switch (request.type()) {
64 case SystemInfoRequest::ADD_REPLICA: {
65 NewReplicaRequest info;
66 if (info.ParseFromString(request.request())) {
67 AddReplica(info.replica_info());
68 }
69 } break;
70 default:
71 break;
72 }
73}
