78// Connect to the server if not connected and send data.
79int NetChannel::SendFromKeepAlive(const std::string& data) {
80 for (int i = 0; i < max_retry_time_; ++i) {
81 if (!long_connecting_) {
82 if (Connect() == 0) {
83 long_connecting_ = true;
84 }
85 if (!long_connecting_) {
86 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
87 continue;
88 }
89 }
90 int ret = SendDataInternal(data);
91 if (ret >= 0) {
92 return ret;
93 }
94 long_connecting_ = false;
95 }
96 return -1;
97}

99int NetChannel::Send(const std::string& data) {
100 if (long_connect_tion_) {
101 return SendFromKeepAlive(data);
102 }
103 for (int i = 0; i < max_retry_time_; ++i) {
104 if (!connected_) {
105 if (Connect()) {
106 LOG(ERROR) << "connect fail:" << ip_ << " port:" << port_;
107 continue;
108 }
109 }
110 int ret = SendDataInternal(data);
111 if (ret >= 0) {
112 return ret;
113 } else if (ret == -1) {
114 LOG(ERROR) << "send data fail:" << ip_ << " port:" << port_;
115 }
116 }
117 return -1;
118}

130// Sign the message if verifier has been provied and send the message to the
131// server.
132std::string NetChannel::GetRawMessageString(
133 const google::protobuf::Message& message, SignatureVerifier* verifier) {
134 ResDBMessage sig_message;
135 if (!message.SerializeToString(sig_message.mutable_data())) {
136 return "";
137 }
138
139 if (verifier != nullptr) {
140 auto signature_or = verifier->SignMessage(sig_message.data());
141 if (!signature_or.ok()) {
142 LOG(ERROR) << "Sign message fail";
143 return "";
144 }
145 sig_message.mutable_signature()->Swap(&(*signature_or));
146 } else {
147 // LOG(ERROR) << " no verifier";
148 }
149
150 std::string message_str;
151 if (!sig_message.SerializeToString(&message_str)) {
152 return "";
153 }
154 return message_str;
155}

165int NetChannel::SendRawMessage(const google::protobuf::Message& message) {
166 ResDBMessage sig_message;
167 if (!message.SerializeToString(sig_message.mutable_data())) {
168 return -1;
169 }
170
171 if (verifier_ != nullptr) {
172 auto signature_or = verifier_->SignMessage(sig_message.data());
173 if (!signature_or.ok()) {
174 LOG(ERROR) << "Sign message fail";
175 return -1;
176 }
177 sig_message.mutable_signature()->Swap(&(*signature_or));
178 }
179 std::string message_str;
180 if (!sig_message.SerializeToString(&message_str)) {
181 return -1;
182 }
183 return Send(message_str);
184}

186int NetChannel::RecvRawMessageStr(std::string* message) {
187 std::string recv_str;
188 int ret = Recv(&recv_str);
189 if (ret <= 0) {
190 return ret;
191 }
192 ResDBMessage sig_message;
193 if (!sig_message.ParseFromString(recv_str)) {
194 LOG(ERROR) << "parse to sig fail";
195 return -1;
196 }
197 *message = sig_message.data();
198 return 0;
199}

201int NetChannel::RecvRawMessage(google::protobuf::Message* message) {
202 std::string resp_data;
203 int ret = Recv(&resp_data);
204 if (ret < 0) {
205 LOG(ERROR) << "recv fail:" << ret;
206 return -1;
207 }
208 ResDBMessage sig_message;
209 if (!sig_message.ParseFromString(resp_data)) {
210 LOG(ERROR) << "parse sig msg fail";
211 return -1;
212 }
213 if (!message->ParseFromString(sig_message.data())) {
214 LOG(ERROR) << "parse response msg fail";
215 return -1;
216 }
217 return 0;
218}
