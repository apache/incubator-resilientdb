29std::unique_ptr<std::string> KVExecutor::ExecuteData(
30 const std::string& request) {
31 KVRequest kv_request;
32 KVResponse kv_response;
33
34 if (!kv_request.ParseFromString(request)) {
35 LOG(ERROR) << "parse data fail";
36 return nullptr;
37 }
38
39 if (kv_request.cmd() == KVRequest::SET) {
40 Set(kv_request.key(), kv_request.value());
41 } else if (kv_request.cmd() == KVRequest::GET) {
42 kv_response.set_value(Get(kv_request.key()));
43 } else if (kv_request.cmd() == KVRequest::GETALLVALUES) {
44 kv_response.set_value(GetAllValues());
45 } else if (kv_request.cmd() == KVRequest::GETRANGE) {
46 kv_response.set_value(GetRange(kv_request.key(), kv_request.value()));
47 } else if (kv_request.cmd() == KVRequest::SET_WITH_VERSION) {
48 SetWithVersion(kv_request.key(), kv_request.value(), kv_request.version());
49 } else if (kv_request.cmd() == KVRequest::GET_WITH_VERSION) {
50 GetWithVersion(kv_request.key(), kv_request.version(),
51 kv_response.mutable_value_info());
52 } else if (kv_request.cmd() == KVRequest::GET_ALL_ITEMS) {
53 GetAllItems(kv_response.mutable_items());
54 } else if (kv_request.cmd() == KVRequest::GET_KEY_RANGE) {
55 GetKeyRange(kv_request.min_key(), kv_request.max_key(),
56 kv_response.mutable_items());
57 } else if (kv_request.cmd() == KVRequest::GET_HISTORY) {
58 GetHistory(kv_request.key(), kv_request.min_version(),
59 kv_request.max_version(), kv_response.mutable_items());
60 } else if (kv_request.cmd() == KVRequest::GET_TOP) {
61 GetTopHistory(kv_request.key(), kv_request.top_number(),
62 kv_response.mutable_items());
63 }
64
65 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
66 if (!kv_response.SerializeToString(resp_str.get())) {
67 return nullptr;
68 }
69 return resp_str;
70}

100void KVExecutor::GetAllItems(Items* items) {
