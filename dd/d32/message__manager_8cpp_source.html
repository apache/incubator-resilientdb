53 checkpoint_manager_->AddCommitData(std::move(request));
54 }
55 },
56 system_info_, std::move(transaction_manager))),
57 collector_pool_(std::make_unique<LockFreeCollectorPool>(
58 "txn", config_.GetMaxProcessTxn(), transaction_executor_.get(),
59 config_.GetConfigData().enable_viewchange())) {
60 global_stats_ = Stats::GetGlobalStats();
61 transaction_executor_->SetSeqUpdateNotifyFunc(
62 [&](uint64_t seq) { collector_pool_->Update(seq - 1); });
63 checkpoint_manager_->SetExecutor(transaction_executor_.get());
64}

108// Check if the request is valid.
109// 1. view is the same as the current view
110// 2. seq is larger or equal than the next execute seq.
111// 3. inside the water mark.
112bool MessageManager::IsValidMsg(const Request& request) {
113 if (request.type() == Request::TYPE_RESPONSE) {
114 return true;
115 }
116 // view should be the same as the current one.
117 if (static_cast<uint64_t>(request.current_view()) != GetCurrentView()) {
118 LOG(ERROR) << "message view :[" << request.current_view()
119 << "] is older than the cur view :[" << GetCurrentView() << "]";
120 return false;
121 }
122
123 if (static_cast<uint64_t>(request.seq()) <
124 transaction_executor_->GetMaxPendingExecutedSeq()) {
125 return false;
126 }
127
128 return true;
129}

131bool MessageManager::MayConsensusChangeStatus(
132 int type, int received_count, std::atomic<TransactionStatue>* status,
133 bool ret) {
134 switch (type) {
135 case Request::TYPE_PRE_PREPARE:
136 if (*status == TransactionStatue::None) {
137 TransactionStatue old_status = TransactionStatue::None;
138 return status->compare_exchange_strong(
139 old_status, TransactionStatue::READY_PREPARE,
140 std::memory_order_acq_rel, std::memory_order_acq_rel);
141 }
142 break;
143 case Request::TYPE_PREPARE:
144 if (*status == TransactionStatue::READY_PREPARE &&
145 config_.GetMinDataReceiveNum() <= received_count) {
146 TransactionStatue old_status = TransactionStatue::READY_PREPARE;
147 return status->compare_exchange_strong(
148 old_status, TransactionStatue::READY_COMMIT,
149 std::memory_order_acq_rel, std::memory_order_acq_rel);
150 }
151 break;
152 case Request::TYPE_COMMIT:
153 if (*status == TransactionStatue::READY_COMMIT &&
154 config_.GetMinDataReceiveNum() <= received_count) {
155 TransactionStatue old_status = TransactionStatue::READY_COMMIT;
156 return status->compare_exchange_strong(
157 old_status, TransactionStatue::READY_EXECUTE,
158 std::memory_order_acq_rel, std::memory_order_acq_rel);
159 return true;
160 }
161 break;
162 }
163 return ret;
164}

166// Add commit messages and return the number of messages have been received.
167// The commit messages only include post(pre-prepare), prepare and commit
168// messages. Messages are handled by state (PREPARE,COMMIT,READY_EXECUTE).
169
170// If there are enough messages and the state is changed after adding the
171// message, return 1, otherwise return 0. Return -2 if the request is not valid.
172CollectorResultCode MessageManager::AddConsensusMsg(
173 const SignatureInfo& signature, std::unique_ptr<Request> request) {
174 if (request == nullptr || !IsValidMsg(*request)) {
175 return CollectorResultCode::INVALID;
176 }
177 int type = request->type();
178 uint64_t seq = request->seq();
179 int resp_received_count = 0;
180 int proxy_id = request->proxy_id();
181
182 int ret = collector_pool_->GetCollector(seq)->AddRequest(
183 std::move(request), signature, type == Request::TYPE_PRE_PREPARE,
184 [&](const Request& request, int received_count,
185 TransactionCollector::CollectorDataType* data,
186 std::atomic<TransactionStatue>* status, bool force) {
187 if (MayConsensusChangeStatus(type, received_count, status, force)) {
188 resp_received_count = 1;
189 }
190 });
191 if (ret == 1) {
192 SetLastCommittedTime(proxy_id);
193 } else if (ret != 0) {
194 return CollectorResultCode::INVALID;
195 }
196 if (resp_received_count > 0) {
197 return CollectorResultCode::STATE_CHANGED;
198 }
199 return CollectorResultCode::OK;
200}

202RequestSet MessageManager::GetRequestSet(uint64_t min_seq, uint64_t max_seq) {
203 RequestSet ret;
204 std::unique_lock<std::mutex> lk(data_mutex_);
205 for (uint64_t i = min_seq; i <= max_seq; ++i) {
206 if (committed_data_.find(i) == committed_data_.end()) {
207 LOG(ERROR) << "seq :" << i << " doesn't exist";
208 continue;
209 }
210 RequestWithProof* request = ret.add_requests();
211 *request->mutable_request() = committed_data_[i];
212 request->set_seq(i);
213 for (const auto& request_info : committed_proof_[i]) {
214 RequestWithProof::RequestData* data = request->add_proofs();
215 *data->mutable_request() = *request_info->request;
216 *data->mutable_signature() = request_info->signature;
217 }
218 }
219 return ret;
220}
