42ResLevelDB::ResLevelDB(std::optional<LevelDBInfo> config) {
43 std::string path = "/tmp/nexres-leveldb";

56void ResLevelDB::CreateDB(const std::string& path) {
57 LOG(ERROR) << "ResLevelDB Create DB: path:" << path
58 << " write buffer size:" << write_buffer_size_
59 << " batch size:" << write_batch_size_;
60 leveldb::Options options;
61 options.create_if_missing = true;
62 options.write_buffer_size = write_buffer_size_;
63
64 leveldb::DB* db = nullptr;
65 leveldb::Status status = leveldb::DB::Open(options, path, &db);
66 if (status.ok()) {
67 db_ = std::unique_ptr<leveldb::DB>(db);
68 }
69 assert(status.ok());
70 LOG(ERROR) << "Successfully opened LevelDB";
71}

79int ResLevelDB::SetValue(const std::string& key, const std::string& value) {
80 batch_.Put(key, value);
81
82 if (batch_.ApproximateSize() >= write_batch_size_) {
83 leveldb::Status status = db_->Write(leveldb::WriteOptions(), &batch_);
84 if (status.ok()) {
85 batch_.Clear();
86 return 0;
87 } else {
88 LOG(ERROR) << "flush buffer fail:" << status.ToString();
89 return -1;
90 }
91 }
92 return 0;
93}

147int ResLevelDB::SetValueWithVersion(const std::string& key,
148 const std::string& value, int version) {
149 std::string value_str = GetValue(key);
150 ValueHistory history;
151 if (!history.ParseFromString(value_str)) {
152 LOG(ERROR) << "old_value parse fail";
153 return -2;
154 }
155
156 int last_v = 0;
157 if (history.value_size() > 0) {
158 last_v = history.value(history.value_size() - 1).version();
159 }
160
161 if (last_v != version) {
162 LOG(ERROR) << "version does not match:" << version
163 << " old version:" << last_v;
164 return -2;
165 }
166
167 Value* new_value = history.add_value();
168 new_value->set_value(value);
169 new_value->set_version(version + 1);
170
171 history.SerializeToString(&value_str);
172 return SetValue(key, value_str);
173}

175std::pair<std::string, int> ResLevelDB::GetValueWithVersion(
176 const std::string& key, int version) {
177 std::string value_str = GetValue(key);
178 ValueHistory history;
179 if (!history.ParseFromString(value_str)) {
180 LOG(ERROR) << "old_value parse fail";
181 return std::make_pair("", 0);
182 }
183 if (history.value_size() == 0) {
184 return std::make_pair("", 0);
185 }
186 if (version > 0) {
187 for (int i = history.value_size() - 1; i >= 0; --i) {
188 if (history.value(i).version() == version) {
189 return std::make_pair(history.value(i).value(),
190 history.value(i).version());
191 }
192 if (history.value(i).version() < version) {
193 break;
194 }
195 }
196 }
197 int last_idx = history.value_size() - 1;
198 return std::make_pair(history.value(last_idx).value(),
199 history.value(last_idx).version());
200}

203std::map<std::string, std::pair<std::string, int>> ResLevelDB::GetAllItems() {
204 std::map<std::string, std::pair<std::string, int>> resp;
205
206 leveldb::Iterator* it = db_->NewIterator(leveldb::ReadOptions());
207 for (it->SeekToFirst(); it->Valid(); it->Next()) {
208 ValueHistory history;
209 if (!history.ParseFromString(it->value().ToString()) ||
210 history.value_size() == 0) {
211 LOG(ERROR) << "old_value parse fail";
212 continue;
213 }
214 const Value& value = history.value(history.value_size() - 1);
215 resp.insert(std::make_pair(it->key().ToString(),
216 std::make_pair(value.value(), value.version())));
217 }
218 delete it;
219
220 return resp;
221}

247 const std::string& key, int min_version, int max_version) {
248 std::vector<std::pair<std::string, int>> resp;
249 std::string value_str = GetValue(key);
250 ValueHistory history;
251 if (!history.ParseFromString(value_str)) {
252 LOG(ERROR) << "old_value parse fail";
253 return resp;
254 }
255
256 for (int i = history.value_size() - 1; i >= 0; --i) {
257 if (history.value(i).version() < min_version) {
258 break;
259 }
260 if (history.value(i).version() <= max_version) {
261 resp.push_back(
262 std::make_pair(history.value(i).value(), history.value(i).version()));
263 }
264 }
265
266 return resp;
267}
