59ReplicaCommunicator::~ReplicaCommunicator() {
60 is_running_ = false;
61 if (broadcast_thread_.joinable()) {
62 broadcast_thread_.join();
63 }
64 if (is_use_long_conn_) {
65 for (auto& cli : client_pools_) {
66 cli.second.reset();
67 }
68 worker_.reset();
69 worker_ = nullptr;
70 io_service_.stop();
71 for (auto& worker_th : worker_threads_) {
72 if (worker_th.joinable()) {
73 worker_th.join();
74 }

111void ReplicaCommunicator::StartBroadcastInBackGround() {
112 is_running_ = true;
113 broadcast_thread_ = std::thread([&]() {
114 while (IsRunning()) {
115 std::vector<std::unique_ptr<QueueItem>> batch_req =
116 batch_queue_.Pop(10000);
117 if (batch_req.empty()) {
118 continue;
119 }
120 BroadcastData broadcast_data;
121 for (auto& queue_item : batch_req) {
122 broadcast_data.add_data()->swap(queue_item->data);
123 }
124
125 global_stats_->SendBroadCastMsg(broadcast_data.data_size());
126 int ret = SendMessageFromPool(broadcast_data, replicas_);
127 if (ret < 0) {
128 LOG(ERROR) << "broadcast request fail:";
129 }
130 }
131 });
132}

134int ReplicaCommunicator::SendMessage(const google::protobuf::Message& message) {
135 global_stats_->BroadCastMsg();
136 if (is_use_long_conn_) {
137 auto item = std::make_unique<QueueItem>();
138 item->data = NetChannel::GetRawMessageString(message, verifier_);
139 batch_queue_.Push(std::move(item));
140 return 0;
141 } else {
142 LOG(ERROR) << "send internal";
143 return SendMessageInternal(message, replicas_);
144 }

178int ReplicaCommunicator::SendMessageFromPool(
179 const google::protobuf::Message& message,
180 const std::vector<ReplicaInfo>& replicas) {
181 int ret = 0;
182 std::string data;
183 message.SerializeToString(&data);
184 global_stats_->SendBroadCastMsgPerRep();
185 std::lock_guard<std::mutex> lk(mutex_);
186 for (const auto& replica : replicas) {
187 auto client = GetClientFromPool(replica.ip(), replica.port());
188 if (client == nullptr) {
189 continue;
190 }
191 if (client->SendMessage(data) == 0) {
192 ret++;
193 } else {
194 LOG(ERROR) << "send to:" << replica.ip() << " fail";
195 }
196 }
197 return ret;
198}

200int ReplicaCommunicator::SendMessageInternal(
201 const google::protobuf::Message& message,
202 const std::vector<ReplicaInfo>& replicas) {
203 int ret = 0;
204 for (const auto& replica : replicas) {
205 auto client = GetClient(replica.ip(), replica.port());
206 if (client == nullptr) {
207 continue;
208 }
209 if (verifier_ != nullptr) {
210 client->SetSignatureVerifier(verifier_);
211 }
212 if (client->SendRawMessage(message) == 0) {
213 ret++;
214 }
215 }
216 return ret;
217}

219AsyncReplicaClient* ReplicaCommunicator::GetClientFromPool(
220 const std::string& ip, int port) {
221 if (client_pools_.find(std::make_pair(ip, port)) == client_pools_.end()) {
222 auto client = std::make_unique<AsyncReplicaClient>(
223 &io_service_, ip, port + (is_use_long_conn_ ? 10000 : 0), true);
224 client_pools_[std::make_pair(ip, port)] = std::move(client);
225 }
226 return client_pools_[std::make_pair(ip, port)].get();
227}

241void ReplicaCommunicator::SendMessage(const google::protobuf::Message& message,
242 int64_t node_id) {
243 ReplicaInfo target_replica;
244 for (const auto& replica : replicas_) {
245 if (replica.id() == node_id) {
246 target_replica = replica;
247 break;
248 }
249 }
250 if (target_replica.ip().empty()) {
251 for (const auto& replica : GetClientReplicas()) {
252 if (replica.id() == node_id) {
253 target_replica = replica;
254 break;
255 }
256 }
257 }
258
259 if (target_replica.ip().empty()) {
260 LOG(ERROR) << "no replica info";
261 return;
262 }
263
264 int ret = SendMessage(message, target_replica);
265 if (ret < 0) {
266 LOG(ERROR) << "broadcast request fail:";
267 }
268}
