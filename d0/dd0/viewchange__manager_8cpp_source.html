56void ViewChangeManager::MayStart() {
57 if (started_) {
58 return;
59 }
60 started_ = true;
61
62 if (config_.GetPublicKeyCertificateInfo()
63 .public_key()
64 .public_key_info()
65 .type() == CertificateKeyInfo::CLIENT) {
66 LOG(ERROR) << "client type not process view change";
67 return;
68 }
69
70 checkpoint_manager_->SetTimeoutHandler([&]() {
71 LOG(ERROR) << "checkpoint timeout";
72 if (status_ == ViewChangeStatus::NONE) {
73 view_change_counter_ = 1;
74 } else if (status_ == ViewChangeStatus::READY_NEW_VIEW) {
75 // If the new view msg expires after receiving enough view
76 // messages, trigger a new primary.
77 view_change_counter_++;
78 }
79 std::lock_guard<std::mutex> lk(status_mutex_);
80 if (ChangeStatue(ViewChangeStatus::READY_VIEW_CHANGE)) {
81 SendViewChangeMsg();
82 }
83 });
84}

101bool ViewChangeManager::IsValidViewChangeMsg(
102 const ViewChangeMessage& view_change_message) {
103 if (view_change_message.view_number() <= system_info_->GetCurrentView()) {
104 LOG(ERROR) << "View number is not greater than current view:"
105 << view_change_message.view_number() << "("
106 << system_info_->GetCurrentView() << ")";
107 return false;
108 }
109
110 if (!checkpoint_manager_->IsValidCheckpointProof(
111 view_change_message.stable_ckpt())) {
112 LOG(ERROR) << "stable checkpoint not invalid";
113 return false;
114 }
115
116 uint64_t stable_seq = view_change_message.stable_ckpt().seq();
117
118 for (const auto& prepared_msg : view_change_message.prepared_msg()) {
119 if (prepared_msg.seq() <= stable_seq) {
120 continue;
121 }
122 // If there is less than 2f+1 proof, reject.
123 if (prepared_msg.proof_size() < config_.GetMinDataReceiveNum()) {
124 LOG(ERROR) << "proof[" << prepared_msg.proof_size()
125 << "] not enough:" << config_.GetMinDataReceiveNum();
126 return false;
127 }
128 for (const auto& proof : prepared_msg.proof()) {
129 if (proof.request().seq() != prepared_msg.seq()) {
130 LOG(ERROR) << "proof seq not match";
131 return false;
132 }
133 std::string data;
134 proof.request().SerializeToString(&data);
135 if (!verifier_->VerifyMessage(data, proof.signature())) {
136 LOG(ERROR) << "proof signature not valid";
137 return false;
138 }
139 }
140 }
141 return true;
142}

167std::vector<std::unique_ptr<Request>> ViewChangeManager::GetPrepareMsg(
168 const NewViewMessage& new_view_message, bool need_sign) {
169 std::map<uint64_t, Request> prepared_msg; // <sequence, digest>
170 for (const auto& msg : new_view_message.viewchange_messages()) {
171 for (const auto& msg : msg.prepared_msg()) {
172 uint64_t seq = msg.seq();
173 prepared_msg[seq] = msg.proof(0).request();
174 }
175 }
176
177 uint64_t min_s = std::numeric_limits<uint64_t>::max();
178 for (const auto& msg : new_view_message.viewchange_messages()) {
179 min_s = std::min(min_s, msg.stable_ckpt().seq());
180 }
181
182 uint64_t max_seq = 1;
183 if (prepared_msg.size() > 0) {
184 max_seq = (--prepared_msg.end())->first;
185 }
186
187 std::vector<std::unique_ptr<Request>> redo_request;
188 // Resent all the request with the current view number.
189 for (auto i = min_s + 1; i <= max_seq; ++i) {
190 if (prepared_msg.find(i) == prepared_msg.end()) {
191 // for sequence hole, create a new request with empty data and
192 // sign by the new primary.
193 std::unique_ptr<Request> client_request = resdb::NewRequest(
194 Request::TYPE_PRE_PREPARE, Request(), config_.GetSelfInfo().id());
195 client_request->set_seq(i);
196 client_request->set_current_view(new_view_message.view_number());
197
198 if (verifier_ && need_sign) {
199 std::string data;
200 auto signature_or = verifier_->SignMessage(data);
201 if (!signature_or.ok()) {
202 LOG(ERROR) << "Sign message fail";
203 continue;
204 }
205 *client_request->mutable_data_signature() = *signature_or;
206 }
207 redo_request.push_back(std::move(client_request));
208 } else {
209 std::unique_ptr<Request> commit_request = resdb::NewRequest(
210 Request::TYPE_COMMIT, prepared_msg[i], config_.GetSelfInfo().id());
211 commit_request->set_seq(i);
212 commit_request->set_current_view(new_view_message.view_number());
213 redo_request.push_back(std::move(commit_request));
214 }
215 }
216
217 return redo_request;
218}

220int ViewChangeManager::ProcessNewView(std::unique_ptr<Context> context,
221 std::unique_ptr<Request> request) {
222 NewViewMessage new_view_message;
223 if (!new_view_message.ParseFromString(request->data())) {
224 LOG(ERROR) << "Parsing new_view_msg failed.";
225 return -2;
226 }
227 LOG(INFO) << "Received NEW-VIEW for view " << new_view_message.view_number();
228 // Check if new view is from next expected primary
229 if (new_view_message.view_number() !=
230 system_info_->GetCurrentView() + view_change_counter_) {
231 LOG(ERROR) << "View number " << new_view_message.view_number()
232 << " is not the same as expected: "
233 << system_info_->GetCurrentView() + 1;
234 return -2;
235 }
236
237 uint64_t min_s = std::numeric_limits<uint64_t>::max(), max_s = 0;
238 // Verify each view change message as the new primary does.
239 for (const auto& msg : new_view_message.viewchange_messages()) {
240 min_s = std::min(min_s, msg.stable_ckpt().seq());
241 max_s = std::max(max_s, msg.stable_ckpt().seq());
242 if (!IsValidViewChangeMsg(msg)) {
243 LOG(ERROR) << "view change message is not invalid";
244 return -2;
245 }
246 }
247
248 // check the re-calculated request is the same as the one in the request.
249 auto request_list = GetPrepareMsg(new_view_message, false);
250 if (request_list.size() !=
251 static_cast<size_t>(new_view_message.request_size())) {
252 LOG(ERROR) << "redo request list size not match:" << request_list.size()
253 << "- " << new_view_message.request_size();
254 return -2;
255 }
256
257 std::set<uint64_t> seq_set;
258 // only check the data.
259 for (size_t i = 0; i < request_list.size(); ++i) {
260 if (request_list[i]->data() != new_view_message.request(i).data()) {
261 LOG(ERROR) << "data not match";
262 return -2;
263 }
264 seq_set.insert(request_list[i]->seq());
265 }
266
267 // Check if all the sequences in the committed list exist.
268 for (uint64_t i = min_s + 1; i <= max_s; i++) {
269 if (seq_set.find(i) == seq_set.end()) {
270 LOG(ERROR) << "Committed msg :" << i << " does exist";
271 return -2;
272 }
273 }
274
275 uint64_t max_seq = *(--seq_set.end());
276 SetCurrentViewAndNewPrimary(new_view_message.view_number());
277 transaction_manager_->SetNextSeq(max_seq + 1);
278
279 // All is fine.
280 for (size_t i = 0; i < request_list.size(); ++i) {
281 LOG(ERROR) << " bc seq:" << request_list[i]->seq();
282 if (new_view_message.request(i).type() ==
283 static_cast<int>(Request::TYPE_PRE_PREPARE)) {
284 replica_client_->SendMessage(new_view_message.request(i),
285 config_.GetSelfInfo());
286 } else {
287 replica_client_->BroadCast(new_view_message.request(i));
288 }
289 }
290
291 ChangeStatue(ViewChangeStatus::NONE);
292 return 0;
293}

295int ViewChangeManager::ProcessViewChange(std::unique_ptr<Context> context,
296 std::unique_ptr<Request> request) {
297 ViewChangeMessage viewchange_message;
298 if (!viewchange_message.ParseFromString(request->data())) {
299 LOG(ERROR) << "pase view change data fail";
300 return -2;
301 }
302
303 if (!IsValidViewChangeMsg(viewchange_message)) {
304 LOG(ERROR) << "view change msg not valid from:" << request->sender_id();
305 return -2;
306 }
307
308 size_t request_size = AddRequest(viewchange_message, request->sender_id());
309 if (request_size >= config_.GetMinClientReceiveNum()) {
310 // process new view
311 if (IsNextPrimary(viewchange_message.view_number())) {
312 std::lock_guard<std::mutex> lk(mutex_);
313 SendNewViewMsg(viewchange_message.view_number());
314 }
315 ChangeStatue(ViewChangeStatus::READY_NEW_VIEW);
316 }
317 return 0;
318}

320void ViewChangeManager::SendNewViewMsg(uint64_t view_number) {
321 if (new_view_is_sent_) {
322 return;
323 }
324 new_view_is_sent_ = true;
325
326 // PBFT Paper - Primary determines the sequence number min-s of the latest
327 // stable checkpoint in V and the highest sequence number max-s in a prepare
328 // message in V
329 // uint64_t min_s = std::numeric_limits<uint64_t>::max();
330 auto requests = viewchange_request_[view_number];
331
332 NewViewMessage new_view_message;
333 new_view_message.set_view_number(view_number);
334
335 std::map<uint64_t, std::string> new_view_request; // <sequence, digest>
336 for (const auto& it : requests) {
337 const ViewChangeMessage& msg = it.second;
338 *new_view_message.add_viewchange_messages() = msg;
339 }
340
341 // Get the redo message from the primary. This could not be done
342 // in each replica because the primary has to signed some empty
343 // request if needed.
344
345 std::vector<std::unique_ptr<Request>> request_list =
346 GetPrepareMsg(new_view_message);
347 for (const auto& request : request_list) {
348 *new_view_message.add_request() = *request;
349 }
350
351 // Broadcast my view change request.
352 std::unique_ptr<Request> request =
353 NewRequest(Request::TYPE_NEWVIEW, Request(), config_.GetSelfInfo().id());
354 new_view_message.SerializeToString(request->mutable_data());
355 replica_client_->BroadCast(*request);
356}

358void ViewChangeManager::SendViewChangeMsg() {
359 // PBFT Paper - <VIEW-CHANGE, v + x, n, C, P)
360 ViewChangeMessage view_change_message;
361 // v + x (view number of the next expected primary)
362 view_change_message.set_view_number(system_info_->GetCurrentView() +
363 view_change_counter_);
364
365 // n (sequence number of the latest checkpoint) and C (proof for the stable
366 // checkpoint)
367 *view_change_message.mutable_stable_ckpt() =
368 checkpoint_manager_->GetStableCheckpointWithVotes();
369
370 // P - P is a set containing a set Pm for each request m that prepared at i
371 // with a sequence number higher than n.
372 int max_seq = checkpoint_manager_->GetMaxTxnSeq();
373 for (int i = view_change_message.stable_ckpt().seq() + 1; i <= max_seq; ++i) {
374 // seq i has been prepared or committed.
375 if (transaction_manager_->GetTransactionState(i) >=
376 TransactionStatue::READY_COMMIT) {
377 std::vector<RequestInfo> proof_info =
378 transaction_manager_->GetPreparedProof(i);
379 auto txn = view_change_message.add_prepared_msg();
380 txn->set_seq(i);
381 for (const auto& info : proof_info) {
382 auto proof = txn->add_proof();
383 *proof->mutable_request() = *info.request;
384 *proof->mutable_signature() = info.signature;
385 }
386 }
387 }
388
389 // Broadcast my view change request.
390 std::unique_ptr<Request> request = NewRequest(
391 Request::TYPE_VIEWCHANGE, Request(), config_.GetSelfInfo().id());
392 view_change_message.SerializeToString(request->mutable_data());
393 replica_client_->BroadCast(*request);
394}
