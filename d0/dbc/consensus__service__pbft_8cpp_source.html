98absl::StatusOr<std::pair<std::unique_ptr<Context>, std::unique_ptr<Request>>>
99ConsensusServicePBFT::PopPendingRequest() {
100 std::lock_guard<std::mutex> lk(mutex_);
101 if (request_pending_.empty()) {
102 LOG(ERROR) << "empty:";
103 return absl::InternalError("No Data.");
104 }
105 auto new_request = std::move(request_pending_.front());
106 request_pending_.pop();
107 return new_request;
108}

110// The implementation of PBFT.
111int ConsensusServicePBFT::ConsensusCommit(std::unique_ptr<Context> context,
112 std::unique_ptr<Request> request) {
113 // LOG(INFO) << "recv impl type:" << request->type() << " "
114 // << "sender id:" << request->sender_id();
115 // If it is in viewchange, push the request to the queue
116 // for the requests from the new view which come before
117 // the local new view done.
118 if (config_.GetConfigData().enable_viewchange()) {
119 view_change_manager_->MayStart();
120 if (view_change_manager_->IsInViewChange()) {
121 switch (request->type()) {
122 case Request::TYPE_NEW_TXNS:
123 case Request::TYPE_PRE_PREPARE:
124 case Request::TYPE_PREPARE:
125 case Request::TYPE_COMMIT:
126 AddPendingRequest(std::move(context), std::move(request));
127 return 0;
128 }
129 } else {
130 while (true) {
131 auto new_request = PopPendingRequest();
132 if (!new_request.ok()) {
133 break;
134 }
135 InternalConsensusCommit(std::move((*new_request).first),
136 std::move((*new_request).second));
137 }
138 }
139 }
140 return InternalConsensusCommit(std::move(context), std::move(request));
141}

143int ConsensusServicePBFT::InternalConsensusCommit(
144 std::unique_ptr<Context> context, std::unique_ptr<Request> request) {
145 // LOG(INFO) << "recv impl type:" << request->type() << " "
146 // << "sender id:" << request->sender_id();
147 switch (request->type()) {
148 case Request::TYPE_CLIENT_REQUEST:
149 if (config_.IsPerformanceRunning()) {
150 return performance_manager_->StartEval();
151 }
152 return response_manager_->NewClientRequest(std::move(context),
153 std::move(request));
154 case Request::TYPE_RESPONSE:
155 if (config_.IsPerformanceRunning()) {
156 return performance_manager_->ProcessResponseMsg(std::move(context),
157 std::move(request));
158 }
159 return response_manager_->ProcessResponseMsg(std::move(context),
160 std::move(request));
161 case Request::TYPE_NEW_TXNS:
162 return commitment_->ProcessNewRequest(std::move(context),
163 std::move(request));
164 case Request::TYPE_PRE_PREPARE:
165 return commitment_->ProcessProposeMsg(std::move(context),
166 std::move(request));
167 case Request::TYPE_PREPARE:
168 return commitment_->ProcessPrepareMsg(std::move(context),
169 std::move(request));
170 case Request::TYPE_COMMIT:
171 return commitment_->ProcessCommitMsg(std::move(context),
172 std::move(request));
173 case Request::TYPE_CHECKPOINT:
174 return checkpoint_manager_->ProcessCheckPoint(std::move(context),
175 std::move(request));
176 case Request::TYPE_VIEWCHANGE:
177 return view_change_manager_->ProcessViewChange(std::move(context),
178 std::move(request));
179 case Request::TYPE_NEWVIEW:
180 return view_change_manager_->ProcessNewView(std::move(context),
181 std::move(request));
182 case Request::TYPE_QUERY:
183 return query_->ProcessQuery(std::move(context), std::move(request));
184 case Request::TYPE_REPLICA_STATE:
185 return query_->ProcessGetReplicaState(std::move(context),
186 std::move(request));
187 }
188 return 0;
189}
