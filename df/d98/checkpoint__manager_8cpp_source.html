86// check whether there are 2f+1 valid checkpoint proof.
87bool CheckPointManager::IsValidCheckpointProof(
88 const StableCheckPoint& stable_ckpt) {
89 std::string hash = stable_ckpt_.hash();
90 std::set<uint32_t> senders;
91 for (const auto& signature : stable_ckpt_.signatures()) {
92 if (!verifier_->VerifyMessage(hash, signature)) {
93 return false;
94 }
95 senders.insert(signature.node_id());
96 }
97 return (senders.size() >= config_.GetMinDataReceiveNum()) ||
98 (stable_ckpt.seq() == 0 && senders.size() == 0);
99}

101int CheckPointManager::ProcessCheckPoint(std::unique_ptr<Context> context,
102 std::unique_ptr<Request> request) {
103 CheckPointData checkpoint_data;
104 if (!checkpoint_data.ParseFromString(request->data())) {
105 LOG(ERROR) << "parse checkpont data fail:";
106 return -2;
107 }
108 uint64_t checkpoint_seq = checkpoint_data.seq();
109 uint32_t sender_id = request->sender_id();
110 int water_mark = config_.GetCheckPointWaterMark();
111 if (checkpoint_seq % water_mark) {
112 LOG(ERROR) << "checkpoint seq not invalid:" << checkpoint_seq;
113 return -2;
114 }
115
116 if (verifier_) {
117 // check signatures
118 bool valid = verifier_->VerifyMessage(checkpoint_data.hash(),
119 checkpoint_data.hash_signature());
120 if (!valid) {
121 LOG(ERROR) << "request is not valid:"
122 << checkpoint_data.hash_signature().DebugString();
123 return -2;
124 }
125 }
126
127 {
128 std::lock_guard<std::mutex> lk(mutex_);
129 auto res =
130 sender_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
131 .insert(sender_id);
132 if (res.second) {
133 sign_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
134 .push_back(checkpoint_data.hash_signature());
135 new_data_++;
136 }
137 Notify();
138 }
139 return 0;
140}

148 int timeout_ms = 1000;
149 std::unique_lock<std::mutex> lk(cv_mutex_);
150 return cv_.wait_for(lk, std::chrono::milliseconds(timeout_ms),
151 [&] { return new_data_ > 0; });

154void CheckPointManager::UpdateStableCheckPointStatus() {
155 while (!stop_) {
156 if (!Wait()) {
157 continue;
158 }
159 uint64_t stable_seq = 0;
160 std::string stable_hash;
161 {
162 std::lock_guard<std::mutex> lk(mutex_);
163 for (auto it : sender_ckpt_) {
164 if (it.second.size() >=
165 static_cast<size_t>(config_.GetMinDataReceiveNum())) {
166 stable_seq = it.first.first;
167 stable_hash = it.first.second;
168 }
169 }
170 new_data_ = 0;
171 }
172
173 LOG(ERROR) << "current stable seq:" << current_stable_seq_
174 << " stable seq:" << stable_seq;
175 std::vector<SignatureInfo> votes;
176 if (current_stable_seq_ < stable_seq) {
177 std::lock_guard<std::mutex> lk(mutex_);
178 votes = sign_ckpt_[std::make_pair(stable_seq, stable_hash)];
179
180 auto it = sender_ckpt_.begin();
181 while (it != sender_ckpt_.end()) {
182 if (it->first.first <= stable_seq) {
183 sign_ckpt_.erase(sign_ckpt_.find(it->first));
184 auto tmp = it++;
185 sender_ckpt_.erase(tmp);
186 } else {
187 it++;
188 }
189 }
190 stable_ckpt_.set_seq(stable_seq);
191 stable_ckpt_.set_hash(stable_hash);
192 for (auto vote : votes) {
193 *stable_ckpt_.add_signatures() = vote;
194 }
195 current_stable_seq_ = stable_seq;
196 LOG(INFO) << "done. stable seq:" << current_stable_seq_
197 << " votes:" << stable_ckpt_.DebugString();
198 }
199 UpdateStableCheckPointCallback(current_stable_seq_);
200 }
201}

218void CheckPointManager::UpdateCheckPointStatus() {
219 uint64_t last_seq = 0;
220 uint64_t last_ckpt_seq = 0;
221 std::string last_hash;
222 int water_mark = config_.GetCheckPointWaterMark();
223 int timeout_ms = config_.GetViewchangeCommitTimeout();
224 while (!stop_) {
225 auto request = data_queue_.Pop(timeout_ms);
226 if (request == nullptr) {
227 if (last_seq > 0) {
228 TimeoutHandler();
229 }
230 continue;
231 }
232 uint64_t current_seq = request->seq();
233 if (current_seq != last_seq + 1) {
234 LOG(ERROR) << "seq invalid:" << last_seq << " current:" << current_seq;
235 continue;
236 }
237 last_hash = GetHash(last_hash, request->hash());
238 last_seq++;
239 txn_db_->Put(std::move(request));
240
241 if (current_seq == last_ckpt_seq + water_mark) {
242 last_ckpt_seq = current_seq;
243 BroadcastCheckPoint(last_ckpt_seq, last_hash);
244 }
245 }
246 return;
247}

249void CheckPointManager::BroadcastCheckPoint(uint64_t seq,
250 const std::string& hash) {
251 CheckPointData checkpoint_data;
252 std::unique_ptr<Request> checkpoint_request = NewRequest(
253 Request::TYPE_CHECKPOINT, Request(), config_.GetSelfInfo().id());
254 checkpoint_data.set_seq(seq);
255 checkpoint_data.set_hash(hash);
256 if (verifier_) {
257 auto signature_or = verifier_->SignMessage(hash);
258 if (!signature_or.ok()) {
259 LOG(ERROR) << "Sign message fail";
260 return;
261 }
262 *checkpoint_data.mutable_hash_signature() = *signature_or;
263 }
264
265 checkpoint_data.SerializeToString(checkpoint_request->mutable_data());
266 replica_communicator_->BroadCast(*checkpoint_request);
267}
