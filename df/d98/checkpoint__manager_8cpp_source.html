90// check whether there are 2f+1 valid checkpoint proof.
91bool CheckPointManager::IsValidCheckpointProof(
92 const StableCheckPoint& stable_ckpt) {
93 std::string hash = stable_ckpt_.hash();
94 std::set<uint32_t> senders;
95 for (const auto& signature : stable_ckpt_.signatures()) {
96 if (!verifier_->VerifyMessage(hash, signature)) {
97 return false;
98 }
99 senders.insert(signature.node_id());
100 }
101
102 return (static_cast<int>(senders.size()) >= config_.GetMinDataReceiveNum()) ||
103 (stable_ckpt.seq() == 0 && senders.size() == 0);
104}

106int CheckPointManager::ProcessCheckPoint(std::unique_ptr<Context> context,
107 std::unique_ptr<Request> request) {
108 CheckPointData checkpoint_data;
109 if (!checkpoint_data.ParseFromString(request->data())) {
110 LOG(ERROR) << "parse checkpont data fail:";
111 return -2;
112 }
113 uint64_t checkpoint_seq = checkpoint_data.seq();
114 uint32_t sender_id = request->sender_id();
115 int water_mark = config_.GetCheckPointWaterMark();
116 if (checkpoint_seq % water_mark) {
117 LOG(ERROR) << "checkpoint seq not invalid:" << checkpoint_seq;
118 return -2;
119 }
120
121 if (verifier_) {
122 // check signatures
123 bool valid = verifier_->VerifyMessage(checkpoint_data.hash(),
124 checkpoint_data.hash_signature());
125 if (!valid) {
126 LOG(ERROR) << "request is not valid:"
127 << checkpoint_data.hash_signature().DebugString();
128 return -2;
129 }
130 }
131
132 {
133 std::lock_guard<std::mutex> lk(mutex_);
134 auto res =
135 sender_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
136 .insert(sender_id);
137 if (res.second) {
138 sign_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
139 .push_back(checkpoint_data.hash_signature());
140 new_data_++;
141 }
142 if (sender_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
143 .size() == 1) {
144 for (auto& hash_ : checkpoint_data.hashs()) {
145 hash_ckpt_[std::make_pair(checkpoint_seq, checkpoint_data.hash())]
146 .push_back(hash_);
147 }
148 }
149 Notify();
150 }
151 return 0;
152}

166void CheckPointManager::UpdateStableCheckPointStatus() {
167 uint64_t last_committable_seq = 0;
168 while (!stop_) {
169 if (!Wait()) {
170 continue;
171 }
172 uint64_t stable_seq = 0;
173 std::string stable_hash;
174 {
175 std::lock_guard<std::mutex> lk(mutex_);
176 for (auto it : sender_ckpt_) {
177 if (it.second.size() >=
178 static_cast<size_t>(config_.GetMinCheckpointReceiveNum())) {
179 committable_seq_ = it.first.first;
180 committable_hash_ = it.first.second;
181 std::set<uint32_t> senders_ =
182 sender_ckpt_[std::make_pair(committable_seq_, committable_hash_)];
183 sem_post(&committable_seq_signal_);
184 if (last_seq_ < committable_seq_ &&
185 last_committable_seq < committable_seq_) {
186 auto replicas_ = config_.GetReplicaInfos();
187 for (auto& replica_ : replicas_) {
188 std::string last_hash;
189 uint64_t last_seq;
190 {
191 std::lock_guard<std::mutex> lk(lt_mutex_);
192 last_hash = last_hash_;
193 // last_seq_ = last_seq > last_committable_seq ? last_seq :
194 // last_committable_seq;
195 last_seq = last_seq_;
196 }
197 if (senders_.count(replica_.id()) &&
198 last_seq < committable_seq_) {
199 // LOG(ERROR) << "GetRequestFromReplica " << last_seq_ + 1 << "
200 // " << committable_seq_;
201 auto requests = txn_accessor_.GetRequestFromReplica(
202 last_seq + 1, committable_seq_, replica_);
203 if (requests.ok()) {
204 bool fail = false;
205 for (auto& request : *requests) {
206 if (SignatureVerifier::CalculateHash(request.data()) !=
207 request.hash()) {
208 LOG(ERROR)
209 << "The hash of the request does not match the data.";
210 fail = true;
211 break;
212 }
213 last_hash = GetHash(last_hash, request.hash());
214 }
215 if (fail) {
216 continue;
217 } else if (last_hash != committable_hash_) {
218 LOG(ERROR) << "The hash of requests returned do not match. "
219 << last_seq + 1 << " " << committable_seq_;
220 } else {
221 last_committable_seq = committable_seq_;
222 for (auto& request : *requests) {
223 if (executor_) {
224 executor_->Commit(std::make_unique<Request>(request));
225 }
226 }
227 SetHighestPreparedSeq(committable_seq_);
228 // LOG(ERROR) << "[4]";
229 break;
230 }
231 }
232 }
233 }
234 }
235 }
236 if (it.second.size() >=
237 static_cast<size_t>(config_.GetMinDataReceiveNum())) {
238 stable_seq = it.first.first;
239 stable_hash = it.first.second;
240 }
241 }
242 new_data_ = 0;
243 }
244
245 // LOG(ERROR) << "current stable seq:" << current_stable_seq_
246 // << " stable seq:" << stable_seq;
247 std::vector<SignatureInfo> votes;
248 if (current_stable_seq_ < stable_seq) {
249 std::lock_guard<std::mutex> lk(mutex_);
250 votes = sign_ckpt_[std::make_pair(stable_seq, stable_hash)];
251 std::set<uint32_t> senders_ =
252 sender_ckpt_[std::make_pair(stable_seq, stable_hash)];
253
254 auto it = sender_ckpt_.begin();
255 while (it != sender_ckpt_.end()) {
256 if (it->first.first <= stable_seq) {
257 sign_ckpt_.erase(sign_ckpt_.find(it->first));
258 auto tmp = it++;
259 sender_ckpt_.erase(tmp);
260 } else {
261 it++;
262 }
263 }
264 stable_ckpt_.set_seq(stable_seq);
265 stable_ckpt_.set_hash(stable_hash);
266 stable_ckpt_.mutable_signatures()->Clear();
267 for (auto vote : votes) {
268 *stable_ckpt_.add_signatures() = vote;
269 }
270 current_stable_seq_ = stable_seq;
271 // LOG(INFO) << "done. stable seq:" << current_stable_seq_
272 // << " votes:" << stable_ckpt_.DebugString();
273 // LOG(INFO) << "done. stable seq:" << current_stable_seq_;
274 }
275 UpdateStableCheckPointCallback(current_stable_seq_);
276 }

290void CheckPointManager::UpdateCheckPointStatus() {
291 uint64_t last_ckpt_seq = 0;
292 int water_mark = config_.GetCheckPointWaterMark();
293 int timeout_ms = config_.GetViewchangeCommitTimeout();
294 std::vector<std::string> stable_hashs;
295 std::vector<uint64_t> stable_seqs;
296 while (!stop_) {
297 auto request = data_queue_.Pop(timeout_ms);
298 if (request == nullptr) {
299 // if (last_seq > 0) {
300 // TimeoutHandler();
301 // }
302 continue;
303 }
304 std::string hash_ = request->hash();
305 uint64_t current_seq = request->seq();
306 if (current_seq != last_seq_ + 1) {
307 LOG(ERROR) << "seq invalid:" << last_seq_ << " current:" << current_seq;
308 continue;
309 }
310 {
311 std::lock_guard<std::mutex> lk(lt_mutex_);
312 last_hash_ = GetHash(last_hash_, request->hash());
313 last_seq_++;
314 }
315 txn_db_->Put(std::move(request));
316
317 if (current_seq == last_ckpt_seq + water_mark) {
318 last_ckpt_seq = current_seq;
319 BroadcastCheckPoint(last_ckpt_seq, last_hash_, stable_hashs, stable_seqs);
320 }
321 }
322 return;
323}

325void CheckPointManager::BroadcastCheckPoint(
326 uint64_t seq, const std::string& hash,
327 const std::vector<std::string>& stable_hashs,
328 const std::vector<uint64_t>& stable_seqs) {
329 CheckPointData checkpoint_data;
330 std::unique_ptr<Request> checkpoint_request = NewRequest(
331 Request::TYPE_CHECKPOINT, Request(), config_.GetSelfInfo().id());
332 checkpoint_data.set_seq(seq);
333 checkpoint_data.set_hash(hash);
334 if (verifier_) {
335 auto signature_or = verifier_->SignMessage(hash);
336 if (!signature_or.ok()) {
337 LOG(ERROR) << "Sign message fail";
338 return;
339 }
340 *checkpoint_data.mutable_hash_signature() = *signature_or;
341 }
342
343 checkpoint_data.SerializeToString(checkpoint_request->mutable_data());
344 replica_communicator_->BroadCast(*checkpoint_request);
345}
