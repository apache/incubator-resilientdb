45template <typename T>
46class AtomicUniquePtr {
47 public:
48 AtomicUniquePtr() : v_(0) {}
49 bool Set(std::unique_ptr<T>& new_ptr) {
50 int old_v = 0;
51 bool res = v_.compare_exchange_strong(old_v, 1, std::memory_order_acq_rel,
52 std::memory_order_acq_rel);
53 if (!res) {
54 return false;
55 }
56 ptr_ = std::move(new_ptr);
57 v_ = 2;
58 return true;
59 }
60
61 T* Reference() {
62 int v = v_.load(std::memory_order_acq_rel);
63 if (v <= 1) {
64 return nullptr;
65 }
66 return ptr_.get();
67 }
68
69 void Clear() {
70 v_ = 0;
71 ptr_ = nullptr;
72 }
73
74 private:
75 std::unique_ptr<T> ptr_;
76 std::atomic<int> v_;
77};
