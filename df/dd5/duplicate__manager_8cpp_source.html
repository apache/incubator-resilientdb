105void DuplicateManager::UpdateRecentHash() {
106 uint64_t time = GetCurrentTime();
107 while (!IsStop()) {
108 time = time + frequency_useconds_;
109 auto sleep_time = time - GetCurrentTime();
110 usleep(sleep_time);
111 while (true) {
112 std::lock_guard<std::mutex> lk(prop_mutex_);
113 if (!proposed_hash_time_queue_.empty()) {
114 auto it = proposed_hash_time_queue_.front();
115 if (it.second + window_useconds_ < time) {
116 proposed_hash_time_queue_.pop();
117 proposed_hash_set_.erase(it.first);
118 } else {
119 break;
120 }
121 } else {
122 break;
123 }
124 }
125
126 while (true) {
127 std::lock_guard<std::mutex> lk(exec_mutex_);
128 if (!executed_hash_time_queue_.empty()) {
129 auto it = executed_hash_time_queue_.front();
130 if (it.second + window_useconds_ < time) {
131 executed_hash_time_queue_.pop();
132 executed_hash_set_.erase(it.first);
133 executed_hash_seq_.erase(it.first);
134 } else {
135 break;
136 }
137 } else {
138 break;
139 }
140 }
141 }
