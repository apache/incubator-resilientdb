64// The implementation of PBFT.
65int ConsensusServicePBFT::ConsensusCommit(std::unique_ptr<Context> context,
66 std::unique_ptr<Request> request) {
67 // LOG(ERROR) << "recv impl type:" << request->type() << " "
68 // << "sender id:" << request->sender_id();
69 switch (request->type()) {
70 case Request::TYPE_CLIENT_REQUEST:
71 if (config_.IsPerformanceRunning()) {
72 return performance_manager_->StartEvaluate();
73 }
74 return response_manager_->NewClientRequest(std::move(context),
75 std::move(request));
76 case Request::TYPE_RESPONSE:
77 return response_manager_->ProcessResponseMsg(std::move(context),
78 std::move(request));
79 case Request::TYPE_NEW_TXNS:
80 return commitment_->ProcessNewRequest(std::move(context),
81 std::move(request));
82 case Request::TYPE_PRE_PREPARE:
83 return commitment_->ProcessProposeMsg(std::move(context),
84 std::move(request));
85 case Request::TYPE_PREPARE:
86 return commitment_->ProcessPrepareMsg(std::move(context),
87 std::move(request));
88 case Request::TYPE_COMMIT:
89 return commitment_->ProcessCommitMsg(std::move(context),
90 std::move(request));
91 case Request::TYPE_RECOVERY_DATA:
92 return recovery_->ProcessRecoveryData(std::move(context),
93 std::move(request));
94 case Request::TYPE_RECOVERY_DATA_RESP:
95 return recovery_->ProcessRecoveryDataResp(std::move(context),
96 std::move(request));
97 case Request::TYPE_CHECKPOINT:
98 return checkpoint_->ProcessCheckPoint(std::move(context),
99 std::move(request));
100 case Request::TYPE_QUERY:
101 return query_->ProcessQuery(std::move(context), std::move(request));
102 case Request::TYPE_REPLICA_STATE:
103 return query_->ProcessGetReplicaState(std::move(context),
104 std::move(request));
105 }
106 return 0;
107}
