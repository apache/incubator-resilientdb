59// Handle the user request and send a pre-prepare message to others.
60// TODO if not a primary, redicet to the primary replica.
61int Commitment::ProcessNewRequest(std::unique_ptr<Context> context,
62 std::unique_ptr<Request> user_request) {
63 if (context == nullptr || context->signature.signature().empty()) {
64 LOG(ERROR) << "user request doesn't contain signature, reject";
65 return -2;
66 }
67
68 if (uint64_t seq =
69 duplicate_manager_->CheckIfExecuted(user_request->hash())) {
70 LOG(ERROR) << "This request is already executed with seq: " << seq;
71 user_request->set_seq(seq);
72 message_manager_->SendResponse(std::move(user_request));
73 return -2;
74 }
75
76 if (config_.GetSelfInfo().id() != message_manager_->GetCurrentPrimary()) {
77 // LOG(ERROR) << "current node is not primary. primary:"
78 // << message_manager_->GetCurrentPrimary()
79 // << " seq:" << user_request->seq()
80 // << " hash:" << user_request->hash();
81 replica_communicator_->SendMessage(*user_request,
82 message_manager_->GetCurrentPrimary());
83 {
84 std::lock_guard<std::mutex> lk(rc_mutex_);
85 request_complained_.push(
86 std::make_pair(std::move(context), std::move(user_request)));
87 }
88
89 return -3;
90 }
91
92 /*
93 if(SignatureVerifier::CalculateHash(user_request->data()) !=
94 user_request->hash()){ LOG(ERROR) << "the hash and data of the user request
95 don't match, reject"; return -2;
96 }
97 */
98
99 // check signatures
100 bool valid = verifier_->VerifyMessage(user_request->data(),
101 user_request->data_signature());
102 if (!valid) {
103 LOG(ERROR) << "request is not valid:"
104 << user_request->data_signature().DebugString();
105 LOG(ERROR) << " msg:" << user_request->data().size();
106 return -2;
107 }
108
109 if (pre_verify_func_ && !pre_verify_func_(*user_request)) {
110 LOG(ERROR) << " check by the user func fail";
111 return -2;
112 }
113
114 global_stats_->IncClientRequest();
115 if (duplicate_manager_->CheckAndAddProposed(user_request->hash())) {
116 LOG(ERROR) << "duplicate check fail:";
117 return -2;
118 }
119 auto seq = message_manager_->AssignNextSeq();
120
121 // Artificially make the primary stop proposing new trasactions.
122
123 if (!seq.ok()) {
124 duplicate_manager_->EraseProposed(user_request->hash());
125 global_stats_->SeqFail();
126 Request request;
127 request.set_type(Request::TYPE_RESPONSE);
128 request.set_sender_id(config_.GetSelfInfo().id());
129 request.set_proxy_id(user_request->proxy_id());
130 request.set_ret(-2);
131 request.set_hash(user_request->hash());
132
133 replica_communicator_->SendMessage(request, request.proxy_id());
134 return -2;
135 }
136
137 user_request->set_type(Request::TYPE_PRE_PREPARE);
138 user_request->set_current_view(message_manager_->GetCurrentView());
139 user_request->set_seq(*seq);
140 user_request->set_sender_id(config_.GetSelfInfo().id());
141 user_request->set_primary_id(config_.GetSelfInfo().id());
142
143 replica_communicator_->BroadCast(*user_request);
144
145 return 0;
146}

148// Receive the pre-prepare message from the primary.
149// TODO check whether the sender is the primary.
150int Commitment::ProcessProposeMsg(std::unique_ptr<Context> context,
151 std::unique_ptr<Request> request) {
152 if (context == nullptr || context->signature.signature().empty()) {
153 LOG(ERROR) << "user request doesn't contain signature, reject";
154 return -2;
155 }
156 if (request->is_recovery()) {
157 if (static_cast<int64_t>(request->seq()) >=
158 message_manager_->GetNextSeq()) {
159 message_manager_->SetNextSeq(request->seq() + 1);
160 }
161 return message_manager_->AddConsensusMsg(context->signature,
162 std::move(request));
163 }
164
165 if (request->sender_id() != message_manager_->GetCurrentPrimary()) {
166 LOG(ERROR) << "the request is not from primary. sender:"
167 << request->sender_id() << " seq:" << request->seq();
168 return -2;
169 }
170
171 /*
172 if(request->hash() != "null" + std::to_string(request->seq())
173 && SignatureVerifier::CalculateHash(request->data()) != request->hash())
174 { LOG(ERROR) << "the hash and data of the request don't match, reject";
175 return -2;
176 }
177 */
178
179 if (request->sender_id() != config_.GetSelfInfo().id()) {
180 if (pre_verify_func_ && !pre_verify_func_(*request)) {
181 LOG(ERROR) << " check by the user func fail";
182 return -2;
183 }
184 BatchUserRequest batch_request;
185 batch_request.ParseFromString(request->data());
186 batch_request.clear_createtime();
187 std::string data;
188 batch_request.SerializeToString(&data);
189 // check signatures
190 bool valid =
191 verifier_->VerifyMessage(request->data(), request->data_signature());
192 if (!valid) {
193 LOG(ERROR) << "request is not valid:"
194 << request->data_signature().DebugString();
195 LOG(ERROR) << " msg:" << request->data().size();
196 return -2;
197 }
198 if (duplicate_manager_->CheckAndAddProposed(request->hash())) {
199 LOG(INFO) << "The request is already proposed, reject";
200 return -2;
201 }
202 }
203
204 global_stats_->IncPropose();
205 std::unique_ptr<Request> prepare_request = resdb::NewRequest(
206 Request::TYPE_PREPARE, *request, config_.GetSelfInfo().id());
207 prepare_request->clear_data();
208
209 // Add request to message_manager.
210 // If it has received enough same requests(2f+1), broadcast the prepare
211 // message.
212 CollectorResultCode ret =
213 message_manager_->AddConsensusMsg(context->signature, std::move(request));
214 if (ret == CollectorResultCode::STATE_CHANGED) {
215 replica_communicator_->BroadCast(*prepare_request);
216 }
217 return ret == CollectorResultCode::INVALID ? -2 : 0;
218}

220// If receive 2f+1 prepare message, broadcast a commit message.
221int Commitment::ProcessPrepareMsg(std::unique_ptr<Context> context,
222 std::unique_ptr<Request> request) {
223 if (context == nullptr || context->signature.signature().empty()) {
224 LOG(ERROR) << "user request doesn't contain signature, reject";
225 return -2;
226 }
227 if (request->is_recovery()) {
228 return message_manager_->AddConsensusMsg(context->signature,
229 std::move(request));
230 }
231 global_stats_->IncPrepare();
232 std::unique_ptr<Request> commit_request = resdb::NewRequest(
233 Request::TYPE_COMMIT, *request, config_.GetSelfInfo().id());
234 commit_request->mutable_data_signature()->Clear();
235 // Add request to message_manager.
236 // If it has received enough same requests(2f+1), broadcast the commit
237 // message.
238 uint64_t seq_ = request->seq();
239 CollectorResultCode ret =
240 message_manager_->AddConsensusMsg(context->signature, std::move(request));
241 if (ret == CollectorResultCode::STATE_CHANGED) {
242 if (message_manager_->GetHighestPreparedSeq() < seq_) {
243 message_manager_->SetHighestPreparedSeq(seq_);
244 }
245 // If need qc, sign the data
246 if (need_qc_ && verifier_) {
247 auto signature_or = verifier_->SignMessage(commit_request->hash());
248 if (!signature_or.ok()) {
249 LOG(ERROR) << "Sign message fail";
250 return -2;
251 }
252 *commit_request->mutable_data_signature() = *signature_or;
253 // LOG(ERROR) << "sign hash"
254 // << commit_request->data_signature().DebugString();
255 }
256 replica_communicator_->BroadCast(*commit_request);
257 }
258 return ret == CollectorResultCode::INVALID ? -2 : 0;
259}

261// If receive 2f+1 commit message, commit the request.
262int Commitment::ProcessCommitMsg(std::unique_ptr<Context> context,
263 std::unique_ptr<Request> request) {
264 if (context == nullptr || context->signature.signature().empty()) {
265 LOG(ERROR) << "user request doesn't contain signature, reject"
266 << " context:" << (context == nullptr);
267 return -2;
268 }
269 if (request->is_recovery()) {
270 return message_manager_->AddConsensusMsg(context->signature,
271 std::move(request));
272 }
273 global_stats_->IncCommit();
274 // Add request to message_manager.
275 // If it has received enough same requests(2f+1), message manager will
276 // commit the request.
277 CollectorResultCode ret =
278 message_manager_->AddConsensusMsg(context->signature, std::move(request));
279 return ret == CollectorResultCode::INVALID ? -2 : 0;
280}

282// =========== private threads ===========================
283// If the transaction is executed, send back to the proxy.
284int Commitment::PostProcessExecutedMsg() {
285 while (!stop_) {
286 auto batch_resp = message_manager_->GetResponseMsg();
287 if (batch_resp == nullptr) {
288 continue;
289 }
290 Request request;
291 request.set_hash(batch_resp->hash());
292 request.set_seq(batch_resp->seq());
293 request.set_type(Request::TYPE_RESPONSE);
294 request.set_sender_id(config_.GetSelfInfo().id());
295 request.set_current_view(batch_resp->current_view());
296 request.set_proxy_id(batch_resp->proxy_id());
297 request.set_primary_id(batch_resp->primary_id());
298 // LOG(ERROR)<<"send back to proxy:"<<batch_resp->proxy_id();
299 batch_resp->SerializeToString(request.mutable_data());
300 replica_communicator_->SendMessage(request, request.proxy_id());
301 }
302 return 0;
303}
