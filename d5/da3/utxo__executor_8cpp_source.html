35std::unique_ptr<std::string> UTXOExecutor::ExecuteData(
36 const std::string& client_request) {
37 UTXORequest utxo_request;
38 UTXOResponse response;
39
40 if (!utxo_request.ParseFromString(client_request)) {
41 LOG(ERROR) << "parse data fail";
42 return nullptr;
43 }
44
45 int64_t ret = transaction_->AddTransaction(utxo_request.utxo());
46 response.set_ret(ret);
47 std::unique_ptr<std::string> ret_str = std::make_unique<std::string>();
48 response.SerializeToString(ret_str.get());
49 return ret_str;
50}

55std::unique_ptr<std::string> QueryExecutor::Query(
56 const std::string& request_str) {
57 UTXOQuery query;
58 if (!query.ParseFromString(request_str)) {
59 LOG(ERROR) << "parse query fail";
60 return nullptr;
61 }
62
63 UTXOQueryResponse resp;
64 if (query.query_transaction()) {
65 std::vector<UTXO> utxos =
66 transaction_->GetUTXO(query.end_id(), query.num());
67 for (const UTXO& utxo : utxos) {
68 *resp.add_utxos() = utxo;
69 }
70 } else {
71 resp.set_value(wallet_->GetCoin(query.address()));
72 }
73 std::unique_ptr<std::string> resp_str = std::make_unique<std::string>();
74 resp.SerializeToString(resp_str.get());
75 return resp_str;
76}
