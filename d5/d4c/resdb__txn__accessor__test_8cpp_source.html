74TEST(ResDBTxnAccessorTest, GetTransactions) {
75 ResDBConfig config({GenerateReplicaInfo(1, "127.0.0.1", 1234),
76 GenerateReplicaInfo(2, "127.0.0.1", 1235),
77 GenerateReplicaInfo(3, "127.0.0.1", 1236),

112TEST(ResDBTxnAccessorTest, GetTransactionsOneDelay) {
113 ResDBConfig config({GenerateReplicaInfo(1, "127.0.0.1", 1234),
114 GenerateReplicaInfo(2, "127.0.0.1", 1235),
115 GenerateReplicaInfo(3, "127.0.0.1", 1236),
116 GenerateReplicaInfo(4, "127.0.0.1", 1237)},
117 GenerateReplicaInfo(1, "127.0.0.1", 1234));
118
119 QueryResponse query_resp;
120 auto txn = query_resp.add_transactions();
121 txn->set_seq(1);
122 txn->set_data("test_resp");
123
124 QueryRequest request;
125 request.set_min_seq(1);
