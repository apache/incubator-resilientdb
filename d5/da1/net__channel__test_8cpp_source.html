168TEST_F(NetChannelTest, SignMessage) {
169 SecretKey my_key = KeyGenerator ::GeneratorKeys(SignatureInfo::RSA);
170 int64_t my_node_id = 1;
171
172 KeyInfo private_key;
173 private_key.set_key(my_key.private_key());
174 private_key.set_hash_type(my_key.hash_type());
175 CertificateInfo cert_info;
176 cert_info.set_node_id(my_node_id);
177 cert_info.mutable_public_key()
178 ->mutable_public_key_info()
179 ->mutable_key()
180 ->set_key(my_key.public_key());
181 cert_info.mutable_public_key()
182 ->mutable_public_key_info()
183 ->mutable_key()
184 ->set_hash_type(my_key.hash_type());
185 cert_info.mutable_public_key()->mutable_public_key_info()->set_node_id(
186 my_node_id);
187
188 SignatureVerifier verifier(private_key, cert_info);
189
190 ClientTestRequest client_request;
191 client_request.set_value("test_value");
192
193 std::string data = GetSendPackageString(client_request);
194
195 std::unique_ptr<MockSocket> socket = std::make_unique<MockSocket>();
196 EXPECT_CALL(*socket, Connect("127.0.0.1", 1234)).WillOnce(Return(0));
197 EXPECT_CALL(*socket, Send).WillOnce(Invoke([&](const std::string& data) {
198 ResDBMessage resdb_message;
199 EXPECT_TRUE(resdb_message.ParseFromString(data));
200 EXPECT_TRUE(verifier.VerifyMessage(resdb_message.data(),
201 resdb_message.signature()));
202 return 0;
203 }));
204
205 NetChannel client("127.0.0.1", 1234);
206 client.SetSocket(std::move(socket));
207 client.SetSignatureVerifier(&verifier);
208
209 EXPECT_EQ(client.SendRawMessage(client_request), 0);
210}
