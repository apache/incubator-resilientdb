39bool RsaVerifyString(const std::string& message, const std::string& public_key,
40 const std::string& signature) {
41 try {
42 // decode and load public key (using pipeline)
43 CryptoPP::RSA::PublicKey rsa_public_key;
44 rsa_public_key.Load(
45 CryptoPP::StringSource(public_key, true, new CryptoPP::HexDecoder())
46 .Ref());
47
48 // decode signature
49 std::string decoded_signature;
50 CryptoPP::StringSource ss(
51 signature, true,
52 new CryptoPP::HexDecoder(new CryptoPP::StringSink(decoded_signature)));
53
54 // verify message
55 bool result = false;
56 CryptoPP::RSASS<CryptoPP::PSSR, CryptoPP::Whirlpool>::Verifier verifier(
57 rsa_public_key);
58 CryptoPP::StringSource ss2(
59 decoded_signature + message, true,
60 new CryptoPP::SignatureVerificationFilter(
61 verifier,
62 new CryptoPP::ArraySink((CryptoPP::byte*)&result, sizeof(result))));
63
64 return result;
65 } catch (...) {
66 LOG(ERROR) << "key not valid";
67 return false;
68 }
69}

101std::string RsaSignString(const std::string& private_key,
102 const std::string& message) {
103 // decode and load private key (using pipeline)
104 CryptoPP::RSA::PrivateKey rsa_private_key;
105 rsa_private_key.Load(
106 CryptoPP::StringSource(private_key, true, new CryptoPP::HexDecoder())
107 .Ref());
108
109 // sign message
110 std::string signature;
111 CryptoPP::RSASS<CryptoPP::PSSR, CryptoPP::Whirlpool>::Signer signer(
112 rsa_private_key);
113 CryptoPP::AutoSeededRandomPool rng;
114 CryptoPP::StringSource ss(
115 message, true,
116 new CryptoPP::SignerFilter(
117 rng, signer,
118 new CryptoPP::HexEncoder(new CryptoPP::StringSink(signature))));
119 return signature;
120}

122std::string ECDSASignString(const std::string& private_key,
123 const std::string& message) {

125 CryptoPP::ECDSA<CryptoPP::ECP, CryptoPP::SHA256>::PrivateKey privateKey;
126 privateKey.Load(
127 CryptoPP::StringSource(private_key, true, new CryptoPP::HexDecoder())
128 .Ref());
129 CryptoPP::AutoSeededRandomPool prng;
130 std::string signature;
131
132 CryptoPP::StringSource ss1(
133 message, true /*pump all*/,
134 new CryptoPP::SignerFilter(
135 prng,
136 CryptoPP::ECDSA<CryptoPP::ECP, CryptoPP::SHA256>::Signer(
137 privateKey),
138 new CryptoPP::StringSink(signature)));
139
140 std::string output;
141 CryptoPP::StringSource(signature, true,
142 new CryptoPP::HexEncoder(
143 new CryptoPP::StringSink(output)) // HexEncoder
144 ); // StringSource
145 return output;
146 } catch (...) {
147 LOG(ERROR) << "sign error";
148 return "";
