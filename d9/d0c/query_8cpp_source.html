55int Query::ProcessQuery(std::unique_ptr<Context> context,
56 std::unique_ptr<Request> request) {
57 QueryRequest query;
58 if (!query.ParseFromString(request->data())) {
59 LOG(ERROR) << "parse data fail";
60 return -2;
61 }
62 // LOG(ERROR) << "request:" << query.DebugString();
63
64 QueryResponse response;
65
66 for (uint64_t i = query.min_seq(); i <= query.max_seq(); ++i) {
67 Request* ret_request = transaction_manager_->GetRequest(i);
68 if (ret_request == nullptr) {
69 break;
70 }
71 Request* txn = response.add_transactions();
72 txn->set_data(ret_request->data());
73 txn->set_seq(ret_request->seq());
74 }
75
76 if (context != nullptr && context->client != nullptr) {
77 // LOG(ERROR) << "send response:" << response.DebugString();
78 int ret = context->client->SendRawMessage(response);
79 if (ret) {
80 LOG(ERROR) << "send resp fail ret:" << ret;
81 }
82 }
83 return 0;
84}
