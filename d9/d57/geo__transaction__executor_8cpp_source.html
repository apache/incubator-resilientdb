50void GeoTransactionExecutor::SendGeoMessages() {
51 std::vector<std::unique_ptr<Request>> messages;
52 while (!IsStop()) {
53 auto message = queue_.Pop(100);
54 if (message != nullptr) {
55 messages.push_back(std::move(message));
56 while (!IsStop() && messages.size() < batch_size_) {
57 auto message = queue_.Pop(0);
58 if (message == nullptr) {
59 break;
60 }
61 messages.push_back(std::move(message));
62 if (messages.size() >= batch_size_) {
63 break;
64 }
65 }
66 }
67 if (messages.size() > 0) {
68 SendBatchGeoMessage(messages);
69 messages.clear();
70 }
71 }
72 return;
73}

75void GeoTransactionExecutor::SendBatchGeoMessage(
76 const std::vector<std::unique_ptr<Request>>& batch_geo_request) {
77 ResConfigData config_data = config_.GetConfigData();
78
79 int self_send = replica_communicator_->SendBatchMessage(
80 batch_geo_request, config_.GetSelfInfo());
81 if (self_send < 0) {
82 LOG(ERROR) << "send batch_geo_request to self FAIL!";
83 }
84 // Only for primary node: send out GEO_REQUEST to other regions.
85 if (config_.GetSelfInfo().id() == system_info_->GetPrimaryId()) {
86 for (const auto& region : config_data.region()) {
87 if (region.region_id() == config_data.self_region_id()) {
88 continue;
89 }
90 // maximum number of faulty replicas in this region
91 int max_faulty = (region.replica_info_size() - 1) / 3;
92 int num_request_sent = 0;
93 for (const auto& replica : region.replica_info()) {
94 // send to f + 1 replicas in the region
95 if (num_request_sent > max_faulty) {
96 break;
97 }
98 int ret =
99 replica_communicator_->SendBatchMessage(batch_geo_request, replica);
100 if (ret >= 0) {
101 num_request_sent++;
102 }
103 }
104 }
105 // LOG(ERROR) << "local executor send out geo message";
106 }
107}

109std::unique_ptr<BatchUserResponse> GeoTransactionExecutor::ExecuteBatch(
110 const BatchUserRequest& request) {
111 std::unique_ptr<Request> geo_request = resdb::NewRequest(
112 Request::TYPE_GEO_REQUEST, Request(), config_.GetSelfInfo().id(),
113 config_.GetConfigData().self_region_id());
114
115 geo_request->set_seq(request.seq());
116 geo_request->set_proxy_id(request.proxy_id());
117 geo_request->set_hash(SignatureVerifier::CalculateHash(
118 geo_request->data() + std::to_string(request.seq()) +
119 std::to_string(config_.GetConfigData().self_region_id())));
120
121 request.SerializeToString(geo_request->mutable_data());
122
123 queue_.Push(std::move(geo_request));
124 return nullptr;
125}
