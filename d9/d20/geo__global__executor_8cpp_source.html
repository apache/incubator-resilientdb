47void GeoGlobalExecutor::Execute(std::unique_ptr<Request> request) {
48 BatchUserRequest batch_request;
49 if (!batch_request.ParseFromString(request->data())) {
50 LOG(ERROR) << "[GeoGlobalExecutor] parse data fail!";
51 return;
52 }
53 if (request->data().empty()) {
54 LOG(ERROR) << "[GeoGlobalExecutor] request->data() is empty ";
55 return;
56 }
57
58 if (global_stats_) {
59 global_stats_->IncTotalGeoRequest(batch_request.user_requests_size());
60 }
61 if (global_transaction_manager_) {
62 auto batch_response =
63 global_transaction_manager_->ExecuteBatch(batch_request);
64 if (batch_response != nullptr &&
65 request->region_info().region_id() == my_region_) {
66 batch_response->set_createtime(batch_request.createtime());
67 batch_response->set_local_id(batch_request.local_id());
68 batch_response->set_proxy_id(batch_request.proxy_id());
69 batch_response->set_seq(batch_request.seq());
70 resp_queue_.Push(std::move(batch_response));
71 }
72 }
73}

82void GeoGlobalExecutor::AddData() {
83 auto request = order_queue_.Pop();
84 if (request == nullptr) {
85 return;
86 }
87 global_stats_->IncGeoRequest();
88 uint64_t seq_num = request->seq();
89 int region_id = request->region_info().region_id();
90 execute_map_[std::make_pair(seq_num, region_id)] = std::move(request);
91}

93std::unique_ptr<Request> GeoGlobalExecutor::GetNextMap() {
94 if (execute_map_.empty()) {
95 return nullptr;
96 }
97 // LOG(ERROR) << "Next round seq: " << next_seq_ << ", current First in map: "
98 // << execute_map_.begin()->first << ", size: " <<
99
100 // LOG(ERROR)<<"get next seq:"<<next_seq_<<" region:"<<next_region_;
101 if (execute_map_.begin()->first == std::make_pair(next_seq_, next_region_)) {
102 std::unique_ptr<Request> res = std::move(execute_map_.begin()->second);
103 execute_map_.erase(execute_map_.begin());
104 next_region_++;
105 if (next_region_ > static_cast<int64_t>(region_size_)) {
106 next_region_ = 1;
107 next_seq_++;
108 }
109 return res;
110 }
111 return nullptr;
112}

114void GeoGlobalExecutor::OrderRound() {
115 while (!IsStop()) {
116 AddData();
117 while (!IsStop()) {
118 std::unique_ptr<Request> seq_map = GetNextMap();
119 if (seq_map == nullptr) {
120 break;
121 }
122 Execute(std::move(seq_map));
123 }
124 }
125}
