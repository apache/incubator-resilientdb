85// Validate transactions committed by the Python SDK
86bool PYVerificator::Validate(const std::string& transaction) {
87 auto locals = py::dict("transaction"_a = transaction);
88
89 py::exec(R"(
90 from sdk_validator.validator import is_valid_tx
91
92 try:
93 txn_dict = json.loads(transaction)
94 ret = is_valid_tx(txn_dict)
95 is_valid = ret[0] == 0
96 except (KeyError, AttributeError, ValueError):
97 is_valid = False
98 )",
99 py::globals(), locals);
100
101 return locals["is_valid"].cast<bool>();
102}
