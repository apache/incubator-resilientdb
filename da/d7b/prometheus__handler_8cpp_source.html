48PrometheusHandler::PrometheusHandler(const std::string& server_address) {
49 exposer_ =
50 prometheus::detail::make_unique<prometheus::Exposer>(server_address);
51
52 registry_ = std::make_shared<prometheus::Registry>();
53 exposer_->RegisterCollectable(registry_);
54
55 Register();
56}
57
58PrometheusHandler::~PrometheusHandler() {
59 if (exposer_) {
60 exposer_->RemoveCollectable(registry_);
61 }
62}

64void PrometheusHandler::Register() {
65 for (auto& table : table_names) {
66 RegisterTable(table.second);
67 }
68
69 for (auto& metric_pair : metric_names) {
70 RegisterMetric(table_names[metric_pair.second.first],
71 metric_pair.second.second);
72 }
73}
74
75void PrometheusHandler::RegisterTable(const std::string& name) {
76 gbuilder* gauge = &prometheus::BuildGauge()
77 .Name(name)
78 .Help(name + " metrics")
79 .Register(*registry_);
80 gauge_[name] = gauge;
81}
82
83void PrometheusHandler::RegisterMetric(const std::string& table_name,
84 const std::string& metric_name) {
85 if (gauge_.find(table_name) == gauge_.end()) {
86 return;
87 }
88 gmetric* metric = &gauge_[table_name]->Add({{"metrics", metric_name}});
89 metric_[metric_name] = metric;
90}
91
92void PrometheusHandler::Set(MetricName name, double value) {
93 std::string metric_name_str = metric_names[name].second;
94 if (metric_.find(metric_name_str) == metric_.end()) {
95 return;
96 }
97 metric_[metric_name_str]->Set(value);
98}

100void PrometheusHandler::Inc(MetricName name, double value) {

103 return;
