29void AppendArgToInput(std::vector<uint8_t>& code, const uint256_t& arg) {
30 const auto pre_size = code.size();

53Address ContractManager::DeployContract(const Address& owner_address,
54 const DeployInfo& deploy_info) {
55 const auto contract_address =
56 AddressManager::CreateContractAddress(owner_address);
57
58 auto contract_constructor = eevm::to_bytes(deploy_info.contract_bin());
59 for (const std::string& param : deploy_info.init_param()) {
60 AppendArgToInput(contract_constructor, param);
61 }
62
63 try {
64 auto contract = gs_->create(contract_address, 0u, contract_constructor);
65
66 auto result = Execute(owner_address, contract_address, {});
67 if (result.ok()) {
68 // set the initialized class context code.
69 contract.acc.set_code(std::move(*result));
70
71 for (const auto& info : deploy_info.func_info()) {
72 SetFuncAddress(contract_address, info);
73 }
74 return contract.acc.get_address();
75 } else {
76 gs_->remove(contract_address);
77 return 0;
78 }
79 } catch (...) {
80 LOG(ERROR) << "Deploy throw expection";
81 return 0;
82 }
83}

94absl::StatusOr<std::string> ContractManager::ExecContract(
95 const Address& caller_address, const Address& contract_address,
96 const Params& func_param) {
97 std::string func_addr =
98 GetFuncAddress(contract_address, func_param.func_name());
99 if (func_addr.empty()) {
100 LOG(ERROR) << "no fouction:" << func_param.func_name();
101 return absl::InvalidArgumentError("Func not exist.");
102 }
103
104 std::vector<uint8_t> inputs = eevm::to_bytes(func_addr);
105 for (const std::string& param : func_param.param()) {
106 AppendArgToInput(inputs, param);
107 }
108
109 auto result = Execute(caller_address, contract_address, inputs);
110
111 if (result.ok()) {
112 return eevm::to_hex_string(*result);
113 }
114 return result.status();
115}

117absl::StatusOr<std::vector<uint8_t>> ContractManager::Execute(
118 const Address& caller_address, const Address& contract_address,
119 const std::vector<uint8_t>& input) {
120 // Ignore any logs produced by this transaction
121 eevm::NullLogHandler ignore;
122 eevm::Transaction tx(caller_address, ignore);
123
124 // Record a trace to aid debugging
125 eevm::Trace tr;
126 eevm::Processor p(*gs_);
127
128 // Run the transaction
129 try {
130 const auto exec_result =
131 p.run(tx, caller_address, gs_->get(contract_address), input, 0u, &tr);
132
133 if (exec_result.er != eevm::ExitReason::returned) {
134 // Print the trace if nothing was returned
135 if (exec_result.er == eevm::ExitReason::threw) {
136 return absl::InternalError(
137 fmt::format("Execution error: {}", exec_result.exmsg));
138 }
139 return absl::InternalError("Deployment did not return");
140 }
141 return exec_result.output;
142 } catch (...) {
143 return absl::InternalError(fmt::format("Execution error:"));
144 }
145}
