74int main(int argc, char** argv) {
75 if (argc < 3) {
76 printf("<cmd> -c [config]\n");
77 return 0;
78 }
79
80 std::string cmd = argv[1];
81 std::string caller_address, contract_name, contract_path, params,
82 contract_address, func_name;
83 int c;
84 std::string client_config_file;
85 while ((c = getopt(argc, argv, "m:c:a:n:p:h:f:s:")) != -1) {
86 switch (c) {
87 case 'm':
88 caller_address = optarg;
89 break;
90 case 'c':
91 client_config_file = optarg;
92 break;
93 case 'n':
94 contract_name = optarg;
95 break;
96 case 'f':
97 func_name = optarg;
98 break;
99 case 'p':
100 contract_path = optarg;
101 break;
102 case 'a':
103 params = optarg;
104 break;
105 case 's':
106 contract_address = optarg;
107 break;
108 case 'h':
109 ShowUsage();
110 break;
111 }
112 }
113
114 printf("cmd = %s config path = %s\n", cmd.c_str(),
115 client_config_file.c_str());
116 ResDBConfig config = GenerateResDBConfig(client_config_file);
117 config.SetClientTimeoutMs(100000);
118
119 ContractClient client(config);
120
121 if (cmd == "create") {
122 CreateAccount(&client);
123 } else if (cmd == "deploy") {
124 std::vector<std::string> init_params;
125 boost::split(init_params, params, boost::is_any_of(","));
126
127 DeployContract(&client, caller_address, contract_name, contract_path,
128 init_params);
129 } else if (cmd == "execute") {
130 printf(
131 "execute\n caller address:%s\n contract address: %s\n func: %s\n "
132 "params:%s\n",
133 caller_address.c_str(), contract_address.c_str(), func_name.c_str(),
134 params.c_str());
135 std::vector<std::string> func_params;
136 boost::split(func_params, params, boost::is_any_of(","));
137
138 ExecuteContract(&client, caller_address, contract_address, func_name,
139 func_params);
140 }
141}
