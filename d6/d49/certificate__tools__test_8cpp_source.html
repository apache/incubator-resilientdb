29CertificateInfo ReadData(const std::string& file_name) {
30 int fd = open(file_name.c_str(), O_RDONLY, 0666);
31 if (fd < 0) {
32 printf("open file %s fail %s\n", file_name.c_str(), strerror(errno));
33 }
34 assert(fd >= 0);
35
36 std::string res;
37 int read_len = 0;
38 char tmp[1024];
39 while (true) {
40 read_len = read(fd, tmp, sizeof(tmp));
41 if (read_len <= 0) {
42 break;
43 }
44 res.append(tmp, read_len);
45 }
46 close(fd);
47 CertificateInfo info;
48 assert(info.ParseFromString(res));
49 return info;
50}
