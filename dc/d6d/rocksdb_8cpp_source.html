42ResRocksDB::ResRocksDB(std::optional<RocksDBInfo> config) {
43 std::string path = "/tmp/nexres-rocksdb";
44 if (config.has_value()) {
45 num_threads_ = (*config).num_threads();
46 write_buffer_size_ = (*config).write_buffer_size_mb() << 20;
47 write_batch_size_ = (*config).write_batch_size();
48 if ((*config).path() != "") {
49 LOG(ERROR) << "Custom path for RocksDB provided in config: "
50 << (*config).path();
51 path = (*config).path();
52 }
53 }
54 CreateDB(path);
55}

57void ResRocksDB::CreateDB(const std::string& path) {
58 rocksdb::Options options;
59 options.create_if_missing = true;
60 if (num_threads_ > 1) options.IncreaseParallelism(num_threads_);
61 options.OptimizeLevelStyleCompaction();
62 options.write_buffer_size = write_buffer_size_;
63
64 rocksdb::DB* db = nullptr;
65 rocksdb::Status status = rocksdb::DB::Open(options, path, &db);
66 if (status.ok()) {
67 db_ = std::unique_ptr<rocksdb::DB>(db);
68 LOG(ERROR) << "Successfully opened RocksDB in path: " << path;
69 } else {
70 LOG(ERROR) << "RocksDB status fail";
71 }
72 assert(status.ok());
73 LOG(ERROR) << "Successfully opened RocksDB";
74}

82int ResRocksDB::SetValue(const std::string& key, const std::string& value) {
83 batch_.Put(key, value);
84
85 if (batch_.Count() >= write_batch_size_) {
86 rocksdb::Status status = db_->Write(rocksdb::WriteOptions(), &batch_);
87 if (status.ok()) {
88 batch_.Clear();
89 } else {
90 LOG(ERROR) << "write value fail:" << status.ToString();
91 return -1;
92 }
93 }
94 return 0;
95}

148int ResRocksDB::SetValueWithVersion(const std::string& key,
149 const std::string& value, int version) {
150 std::string value_str = GetValue(key);
151 ValueHistory history;
152 if (!history.ParseFromString(value_str)) {
153 LOG(ERROR) << "old_value parse fail";
154 return -2;
155 }
156
157 int last_v = 0;
158 if (history.value_size() > 0) {
159 last_v = history.value(history.value_size() - 1).version();
160 }
161
162 if (last_v != version) {
163 LOG(ERROR) << "version does not match:" << version
164 << " old version:" << last_v;
165 return -2;
166 }
167
168 Value* new_value = history.add_value();
169 new_value->set_value(value);
170 new_value->set_version(version + 1);
171
172 history.SerializeToString(&value_str);
173 return SetValue(key, value_str);
174}

176std::pair<std::string, int> ResRocksDB::GetValueWithVersion(
177 const std::string& key, int version) {
178 std::string value_str = GetValue(key);
179 ValueHistory history;
180 if (!history.ParseFromString(value_str)) {
181 LOG(ERROR) << "old_value parse fail";
182 return std::make_pair("", 0);
183 }
184 if (history.value_size() == 0) {
185 return std::make_pair("", 0);
186 }
187 if (version > 0) {
188 for (int i = history.value_size() - 1; i >= 0; --i) {
189 if (history.value(i).version() == version) {
190 return std::make_pair(history.value(i).value(),
191 history.value(i).version());
192 }
193 if (history.value(i).version() < version) {
194 break;
195 }
196 }
197 }
198 int last_idx = history.value_size() - 1;
199 return std::make_pair(history.value(last_idx).value(),
200 history.value(last_idx).version());
201}

204std::map<std::string, std::pair<std::string, int>> ResRocksDB::GetAllItems() {
205 std::map<std::string, std::pair<std::string, int>> resp;
206
207 rocksdb::Iterator* it = db_->NewIterator(rocksdb::ReadOptions());
208 for (it->SeekToFirst(); it->Valid(); it->Next()) {
209 ValueHistory history;
210 if (!history.ParseFromString(it->value().ToString()) ||
211 history.value_size() == 0) {
212 LOG(ERROR) << "old_value parse fail";
213 continue;
214 }
215 const Value& value = history.value(history.value_size() - 1);
216 resp.insert(std::make_pair(it->key().ToString(),
217 std::make_pair(value.value(), value.version())));
218 }
219 delete it;
220
221 return resp;
222}

248 const std::string& key, int min_version, int max_version) {
249 std::vector<std::pair<std::string, int>> resp;
250 std::string value_str = GetValue(key);
251 ValueHistory history;

253 LOG(ERROR) << "old_value parse fail";
254 return resp;
255 }
256
257 for (int i = history.value_size() - 1; i >= 0; --i) {
258 if (history.value(i).version() < min_version) {
259 break;
260 }
261 if (history.value(i).version() <= max_version) {
262 resp.push_back(
263 std::make_pair(history.value(i).value(), history.value(i).version()));
264 }
265 }
266
