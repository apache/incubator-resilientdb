179TEST_F(CheckPointManagerTest, StableCkpt) {
180 config_.SetViewchangeCommitTimeout(100);
181 std::promise<bool> ckp_done;
182 std::future<bool> ckp_done_future = ckp_done.get_future();
183 bool already_set = 0;
184 MyCheckPointManager manager(config_, &replica_communicator_, nullptr,
185 [&](int64_t seq) {
186 if (seq > 0 && !already_set) {
187 already_set = true;
188 ckp_done.set_value(true);
189 }
190 });
191 for (int i = 1; i <= 4; ++i) {
192 CheckPointData checkpoint_data;
193 std::unique_ptr<Request> checkpoint_request =
194 NewRequest(Request::TYPE_CHECKPOINT, Request(), i);
195 checkpoint_data.set_seq(5);
196 checkpoint_data.set_hash("1234");
197 checkpoint_data.SerializeToString(checkpoint_request->mutable_data());
198 EXPECT_EQ(manager.ProcessCheckPoint(std::make_unique<Context>(),
199 std::move(checkpoint_request)),
200 0);
201 }
202 ckp_done_future.get();
203 EXPECT_EQ(manager.GetStableCheckpoint(), 5);
204}

249TEST_F(CheckPointManagerTest, Votes) {
250 config_.SetViewchangeCommitTimeout(100);
251 MockSignatureVerifier mock_verifier;
252 EXPECT_CALL(mock_verifier, SignMessage).WillOnce(Return(SignatureInfo()));
253 EXPECT_CALL(mock_verifier, VerifyMessage(_, EqualsProto(SignatureInfo())))
254 .WillRepeatedly(Return(true));
255
256 std::promise<bool> propose_done;
257 std::future<bool> propose_done_future = propose_done.get_future();
258 CheckPointManager manager(config_, &replica_communicator_, &mock_verifier);
259 EXPECT_CALL(replica_communicator_, BroadCast)
260 .WillRepeatedly(Invoke([&](const google::protobuf::Message& message) {
261 for (int i = 1; i <= 3; ++i) {
262 std::unique_ptr<Request> checkpoint_request =
263 std::make_unique<Request>();
264 checkpoint_request->CopyFrom(message);
265 checkpoint_request->set_sender_id(i);
266 EXPECT_EQ(manager.ProcessCheckPoint(std::make_unique<Context>(),
267 std::move(checkpoint_request)),
268 0);
269 }
270 propose_done.set_value(true);
271 }));
272
273 std::unique_ptr<Request> request = std::make_unique<Request>();
274 for (int i = 1; i <= 5; ++i) {
275 std::unique_ptr<Request> request = std::make_unique<Request>();
276 request->set_seq(i);
277 manager.AddCommitData(std::move(request));
278 }
279 propose_done_future.get();
280 sleep(1);
281 StableCheckPoint ckpt = manager.GetStableCheckpointWithVotes();
282 EXPECT_EQ(ckpt.seq(), 5);
283 EXPECT_EQ(ckpt.signatures_size(), 3);
284}
